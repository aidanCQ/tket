43static Circuit multiplexor_sequential_decomp(
44 const ctrl_op_map_t &op_map, unsigned n_controls, unsigned n_targets) {
45 Circuit c(n_controls + n_targets);
46 std::vector<unsigned> qubits(n_controls + n_targets);
47 std::iota(std::begin(qubits), std::end(qubits), 0);
48 for (auto it = op_map.begin(); it != op_map.end(); it++) {
49 std::vector<unsigned> zero_ctrls;
50 for (unsigned i = 0; i < it->first.size(); i++) {
51 if (!it->first[i]) {
52 zero_ctrls.push_back(i);
53 c.add_op<unsigned>(OpType::X, {i});
54 }
55 }
56 QControlBox qcbox(it->second, n_controls);
57 c.add_box(qcbox, qubits);
58 for (unsigned i : zero_ctrls) {
59 c.add_op<unsigned>(OpType::X, {i});
60 }
61 }
62 return c;
63}

70enum class RecursionNodeType { Left = 0, Right = 1, Root = 2 };

101static void recursive_demultiplex_rotation(
102 const std::vector<Expr> &angles, const OpType &axis, unsigned total_qubits,
103 Circuit &circ, const RecursionNodeType &node_type) {
104 unsigned n_rotations = angles.size();
105 unsigned n_qubits = (unsigned)log2(n_rotations) + 1;
106 unsigned mid = (unsigned)(n_rotations / 2);
107 std::vector<Expr> p_angles;
108 std::vector<Expr> q_angles;
109 for (unsigned i = 0; i < mid; i++) {
110 p_angles.push_back((angles[i] - angles[mid + i]) / 2);
111 q_angles.push_back((angles[i] + angles[mid + i]) / 2);
112 }
113 // UCR = CX P CX Q = Q CX P CX
114 // the left recursion child implements P CX Q, and the
115 // right recursion child implements Q CX P to cancel the CXs
116 if (node_type == RecursionNodeType::Right) {
117 std::swap(p_angles, q_angles);
118 }
119 if (q_angles.size() == 1) {
120 // base step
121 circ.add_op<unsigned>(axis, q_angles[0], {total_qubits - 1});
122 } else {
123 recursive_demultiplex_rotation(
124 q_angles, axis, total_qubits, circ, RecursionNodeType::Left);
125 }
126 circ.add_op<unsigned>(
127 OpType::CX, {total_qubits - n_qubits, total_qubits - 1});
128 if (p_angles.size() == 1) {
129 // base step
130 circ.add_op<unsigned>(axis, p_angles[0], {total_qubits - 1});
131 } else {
132 recursive_demultiplex_rotation(
133 p_angles, axis, total_qubits, circ, RecursionNodeType::Right);
134 }
135 if (node_type == RecursionNodeType::Root) {
136 // for the root step, we implement UCR = CX P CX Q
137 circ.add_op<unsigned>(
138 OpType::CX, {total_qubits - n_qubits, total_qubits - 1});
139 }
140}

152static std::tuple<Eigen::Matrix2cd, Eigen::Matrix2cd, double, double>
153constant_demultiplex(const Eigen::Matrix2cd &a, const Eigen::Matrix2cd &b) {
154 Eigen::Matrix2cd X = a * b.adjoint();
155 // decompose X using eq(19)
156 std::vector<double> tk1_params = tk1_angles_from_unitary(X);
157 Eigen::Matrix2cd tk1_su2 = get_matrix_from_tk1_angles(
158 {tk1_params[0], tk1_params[1], tk1_params[2], 0.0});
159 Complex x0 = tk1_su2(0, 0);
160 double phi = tk1_params[3] * PI * 2;
161 // compute r matrix using eq(11) and eq(12)
162 double a0 = -PI / 2 - phi / 2 - std::arg(x0);
163 double a1 = PI / 2 - phi / 2 + std::arg(x0);
164 Complex r0 = std::exp(0.5 * i_ * a0);
165 Complex r1 = std::exp(0.5 * i_ * a1);
166 Eigen::Matrix2cd r = Eigen::Matrix2cd::Zero();
167 r(0, 0) = r0;
168 r(1, 1) = r1;
169 // a0 and a1 defines the R matrix in eq(3). And they are the z rotation angles
170 // activated by 0 and 1 respectively.
171 a0 = a0 / PI;
172 a1 = a1 / PI;
173 Eigen::ComplexEigenSolver<Eigen::Matrix2cd> eigen_solver(r * X * r);
174 Eigen::Matrix2cd u = eigen_solver.eigenvectors();
175 // the eigenvalues are guaranteed to be {i, -i}. We permute u so its
176 // eigenvalues have the order [i, -i].
177 if (std::abs(eigen_solver.eigenvalues()[0] + i_) < EPS) {
178 u.col(0).swap(u.col(1));
179 }
180 Eigen::Matrix2cd d = Eigen::Matrix2cd::Zero();
181 d(0, 0) = std::sqrt(i_);
182 d(1, 1) = std::sqrt(-i_);
183 Eigen::Matrix2cd v = d * u.adjoint() * r.adjoint() * b;
184 return {u, v, a0, a1};
185}

194static std::vector<Eigen::Matrix2cd> ucrz_angles_to_diagonal(
195 const std::vector<double> &angles) {
196 std::vector<Eigen::Matrix2cd> diag;
197 unsigned mid = (unsigned)(angles.size() / 2);
198 for (unsigned i = 0; i < mid; i++) {
199 Eigen::Matrix2cd u = Eigen::Matrix2cd::Zero();
200 u(0, 0) = std::exp(-0.5 * angles[i] * i_ * PI);
201 u(1, 1) = std::exp(-0.5 * angles[i + mid] * i_ * PI);
202 diag.push_back(u);
203 }
204 for (unsigned i = 0; i < mid; i++) {
205 Eigen::Matrix2cd u = Eigen::Matrix2cd::Zero();
206 u(0, 0) = std::exp(0.5 * angles[i] * i_ * PI);
207 u(1, 1) = std::exp(0.5 * angles[i + mid] * i_ * PI);
208 diag.push_back(u);
209 }
210 return diag;
211}
243static void recursive_demultiplex_u2(
244 std::vector<Eigen::Matrix2cd> &unitaries, unsigned total_qubits,
245 Circuit &circ, std::vector<std::vector<double>> &ucrzs,
246 const Eigen::Matrix2cd &left_compose,
247 const Eigen::Matrix2cd &right_compose) {
248 // The following two constant matrices are the bottom SQ unitaries resulted
249 // from decomposing the D gate (i.e. ZZPhase(-0.5)) using CX
250 const static Eigen::Matrix2cd U_MULT =
251 get_matrix_from_tk1_angles({0.5, 0.5, 0.5, 0.0});
252 const static Eigen::Matrix2cd V_MULT =
253 get_matrix_from_tk1_angles({0.5, 0.5, 0, 0.0});
254 unsigned n_unitaries = unitaries.size();
255 unsigned n_qubits = (unsigned)log2(n_unitaries) + 1;
256 unsigned mid = (unsigned)(n_unitaries / 2);
257 // We generalise eq(3) for n controls, demultiplex the multiplexor
258 // by demultiplexing all pairs {unitaries[i], unitaries[mid+i]} 0<=i<mid.
259 // i.e. I tensor diag(u) = I tensor diag(u_list)
260 // I tensor diag(v) = I tensor diag(v_list)
261 // D = ZZPhase(-0.5)
262 // R = UCRz(rz_list, [q_{n-1}, q_{1}, q_{2}, ..., q_{n-2}, q_0])
263 std::vector<Eigen::Matrix2cd> u_list;
264 std::vector<Eigen::Matrix2cd> v_list;
265 std::vector<double> rz_list(n_unitaries);
266
267 // merge previous UCRz gate into the multiplexor
268 std::vector<Eigen::Matrix2cd> ucrz_diag =
269 ucrz_angles_to_diagonal(ucrzs[n_qubits - 2]);
270 for (unsigned i = 0; i < unitaries.size(); i++) {
271 unitaries[i] = unitaries[i] * ucrz_diag[i];
272 }
273 // demultiplex pairs (unitaries[i], unitaries[mid+i])
274 for (unsigned i = 0; i < mid; i++) {
275 auto [u, v, a0, a1] =
276 constant_demultiplex(unitaries[i], unitaries[mid + i]);
277 u_list.push_back(u);
278 v_list.push_back(v);
279 rz_list[i] = a0;
280 rz_list[i + mid] = a1;
281 }
282
283 // update the ucrzs with the 1.5 angle resulted from decomposing ZZPhase(-0.5)
284 std::for_each(rz_list.begin(), rz_list.end(), [](double &f) { f += 1.5; });
285 ucrzs[n_qubits - 2] = rz_list;
286
287 // adding gates to the circuit
288 // add v
289 if (v_list.size() == 1) {
290 Eigen::Matrix2cd v_prime = V_MULT * v_list[0] * left_compose;
291 circ.add_box(Unitary1qBox(v_prime), {total_qubits - 1});
292 } else {
293 recursive_demultiplex_u2(
294 v_list, total_qubits, circ, ucrzs, left_compose, V_MULT);
295 }
296 // add CX
297 circ.add_op<unsigned>(
298 OpType::CX, {total_qubits - n_qubits, total_qubits - 1});
299 circ.add_phase(1.75);
300 // add u
301 if (u_list.size() == 1) {
302 Eigen::Matrix2cd u_prime = right_compose * u_list[0] * U_MULT;
303 circ.add_box(Unitary1qBox(u_prime), {total_qubits - 1});
304 } else {
305 recursive_demultiplex_u2(
306 u_list, total_qubits, circ, ucrzs, U_MULT, right_compose);
307 }
308 return;
309}

311static void op_map_validate(const ctrl_op_map_t &op_map) {
312 unsigned n_controls = 0;
313 unsigned n_targets = 0;
314 for (auto it = op_map.begin(); it != op_map.end(); it++) {
315 op_signature_t op_sig = it->second->get_signature();
316 if ((unsigned long)std::count(
317 op_sig.begin(), op_sig.end(), EdgeType::Quantum) != op_sig.size()) {
318 throw BadOpType(
319 "Multiplexed operations cannot have classical wires.",
320 it->second->get_type());
321 }
322 if (it == op_map.begin()) {
323 n_controls = (unsigned)it->first.size();
324 n_targets = (unsigned)op_sig.size();
325 } else {
326 if (it->first.size() != n_controls) {
327 throw std::invalid_argument(
328 "The bitstrings passed to the multiplexor must have the same "
329 "width.");
330 }
331 if (op_sig.size() != n_targets) {
332 throw std::invalid_argument(
333 "Multiplexed operations must have the same width.");
334 }
335 }
336 }
337}

339static ctrl_op_map_t op_map_symbol_sub(
340 const SymEngine::map_basic_basic &sub_map, const ctrl_op_map_t &op_map) {
341 ctrl_op_map_t new_op_map;
342 for (auto it = op_map.begin(); it != op_map.end(); it++) {
343 new_op_map.insert({it->first, it->second->symbol_substitution(sub_map)});
344 }
345 return new_op_map;

348static SymSet op_map_free_symbols(const ctrl_op_map_t &op_map) {
349 SymSet all_symbols;
350 for (auto it = op_map.begin(); it != op_map.end(); it++) {
351 SymSet op_symbols = it->second->free_symbols();
352 all_symbols.insert(op_symbols.begin(), op_symbols.end());
353 }
354 return all_symbols;

373static bool opmap_it_equal(
374 const std::pair<std::vector<bool>, Op_ptr> &lhs,
375 const std::pair<std::vector<bool>, Op_ptr> &rhs) {
376 return lhs.first == rhs.first && *lhs.second == *rhs.second;
377}
378static bool tensored_opmap_it_equal(
379 const std::pair<std::vector<bool>, std::vector<Op_ptr>> &lhs,
380 const std::pair<std::vector<bool>, std::vector<Op_ptr>> &rhs) {
381 return lhs.first == rhs.first &&
382 std::equal(
383 lhs.second.begin(), lhs.second.end(), rhs.second.begin(),
384 rhs.second.end(),
385 [](const Op_ptr &a, const Op_ptr &b) { return *a == *b; });
386}

388// Check if two ctrl_op_map_t are semantically equal.
389static bool opmap_compare(
390 const ctrl_op_map_t &map1, const ctrl_op_map_t &map2) {
391 return std::equal(
392 map1.begin(), map1.end(), map2.begin(), map2.end(), opmap_it_equal);
393}
394// Check if two ctrl_tensored_op_map_t are semantically equal.
395static bool opmap_compare(
396 const ctrl_tensored_op_map_t &map1, const ctrl_tensored_op_map_t &map2) {
397 return std::equal(
398 map1.begin(), map1.end(), map2.begin(), map2.end(),
399 tensored_opmap_it_equal);

468MultiplexedRotationBox::MultiplexedRotationBox(const ctrl_op_map_t &op_map)
469 : Box(OpType::MultiplexedRotationBox), op_map_(op_map) {
470 auto it = op_map.begin();
471 if (it == op_map.end()) {
472 throw std::invalid_argument(
473 "The op_map argument passed to MultiplexedRotationBox cannot be "
474 "empty.");
475 }
476 for (; it != op_map.end(); it++) {
477 if (it == op_map.begin()) {
478 n_controls_ = (unsigned)it->first.size();
479 if (n_controls_ > MAX_N_CONTROLS) {
480 throw std::invalid_argument(
481 "MultiplexedRotationBox only supports bitstrings up to " +
482 std::to_string(MAX_N_CONTROLS) + " bits.");
483 }
484 axis_ = it->second->get_type();
485 if (axis_ != OpType::Rx && axis_ != OpType::Ry && axis_ != OpType::Rz) {
486 throw BadOpType(
487 "Ops passed to MultiplexedRotationBox must be either Rx, Ry, or "
488 "Rz.",
489 axis_);
490 }
491 } else {
492 if (it->second->get_type() != axis_) {
493 throw std::invalid_argument(
494 "Ops passed to MultiplexedRotationBox must have the same rotation "
495 "type.");
496 }
497 }
498 }
499 op_map_validate(op_map);
500}

554void MultiplexedRotationBox::generate_circuit() const {
555 Circuit circ(n_controls_ + 1);
556 if (n_controls_ == 0) {
557 auto it = op_map_.begin();
558 circ.add_op<unsigned>(it->second, {0});
559 circ_ = std::make_shared<Circuit>(circ);
560 return;
561 }
562 unsigned long long n_rotations = 1ULL << n_controls_;
563 std::vector<Expr> rotations(n_rotations);
564 // convert op_map to a vector of 2^n_controls_ angles
565 for (unsigned long long i = 0; i < n_rotations; i++) {
566 auto it = op_map_.find(dec_to_bin(i, n_controls_));
567 if (it == op_map_.end()) {
568 rotations[i] = 0;
569 } else {
570 rotations[i] = it->second->get_params()[0];
571 }
572 }
573 OpType axis = axis_;
574 if (axis_ == OpType::Rx) {
575 circ.add_op<unsigned>(OpType::H, {n_controls_});
576 axis = OpType::Rz;
577 }
578 recursive_demultiplex_rotation(
579 rotations, axis, n_controls_ + 1, circ, RecursionNodeType::Root);
580 if (axis_ == OpType::Rx) {
581 circ.add_op<unsigned>(OpType::H, {n_controls_});
582 }
583 circ_ = std::make_shared<Circuit>(circ);
584}

586MultiplexedU2Box::MultiplexedU2Box(const ctrl_op_map_t &op_map, bool impl_diag)
587 : Box(OpType::MultiplexedU2Box), op_map_(op_map), impl_diag_(impl_diag) {
588 auto it = op_map.begin();
589 if (it == op_map.end()) {
590 throw std::invalid_argument(
591 "The op_map argument passed to MultiplexedU2Box cannot be empty.");
592 }
593 n_controls_ = (unsigned)it->first.size();
594 if (n_controls_ > MAX_N_CONTROLS) {
595 throw std::invalid_argument(
596 "MultiplexedU2Box only supports bitstrings up to " +
597 std::to_string(MAX_N_CONTROLS) + " bits.");
598 }
599 for (; it != op_map.end(); it++) {
600 OpType optype = it->second->get_type();
601 if (!is_single_qubit_unitary_type(optype) &&
602 optype != OpType::Unitary1qBox) {
603 throw BadOpType(
604 "Ops passed to MultiplexedU2Box must be single-qubit unitary gate "
605 "types or Unitary1qBox.",
606 optype);
607 }
608 }
609 op_map_validate(op_map);
610}

666std::pair<Circuit, Eigen::VectorXcd> MultiplexedU2Box::decompose() const {
667 Circuit circ(n_controls_ + 1);
668 if (n_controls_ == 0) {
669 auto it = op_map_.begin();
670 circ.add_op<unsigned>(it->second, {0});
671 return std::make_pair(circ, Eigen::VectorXcd::Constant(2, 1));
672 }
673 unsigned long long n_unitaries = 1ULL << n_controls_;
674 std::vector<Eigen::Matrix2cd> unitaries(n_unitaries);
675 // convert op_map to a vector of 2^n_controls_ unitaries
676 for (unsigned long long i = 0; i < n_unitaries; i++) {
677 auto it = op_map_.find(dec_to_bin(i, n_controls_));
678 if (it == op_map_.end()) {
679 unitaries[i] = Eigen::Matrix2cd::Identity();
680 } else {
681 if (it->second->get_type() == OpType::Unitary1qBox) {
682 std::shared_ptr<const Unitary1qBox> u1box =
683 std::dynamic_pointer_cast<const Unitary1qBox>(it->second);
684 unitaries[i] = u1box->get_matrix();
685 } else {
686 if (!it->second->free_symbols().empty()) {
687 throw Unsupported("Can't decompose symbolic MultiplexedU2Box.");
688 }
689 unitaries[i] = GateUnitaryMatrix::get_unitary(*as_gate_ptr(it->second));
690 }
691 }
692 }
693 // initialise the ucrz list
694 std::vector<std::vector<double>> ucrzs(n_controls_);
695 for (unsigned i = 0; i < n_controls_; i++) {
696 ucrzs[i] = std::vector<double>(1ULL << (i + 1), 0.0);
697 }
698 recursive_demultiplex_u2(
699 unitaries, n_controls_ + 1, circ, ucrzs, Eigen::Matrix2cd::Identity(),
700 Eigen::Matrix2cd::Identity());
701 // convert the ucrzs to a diagonal matrix
702 Eigen::VectorXcd diag =
703 Eigen::VectorXcd::Constant(1ULL << (n_controls_ + 1), 1);
704 for (unsigned i = 0; i < n_controls_; i++) {
705 // ith ucrzs acts on i+2 qubits
706 // which has n_controls_ + 1 - (i+2) identities its the tensor product
707 // therefore (n_controls_ + 1 - (i+2))^2 copies in the diagonal
708 for (unsigned long long offset = 0;
709 offset < (1ULL << (n_controls_ + 1 - (i + 2))); offset++) {
710 for (unsigned long long j = 0; j < (1ULL << (i + 1)); j++) {
711 // the bitstrings in a ucrz are mapped to qubits not in the standard
712 // order
713 unsigned long long diag_idx =
714 (j >= (1ULL << i)) ? (j - (1ULL << i)) * 2 + 1 : j * 2;
715 diag[diag_idx + offset * (1ULL << (i + 2))] *=
716 std::exp(-0.5 * i_ * PI * ucrzs[i][j]);
717 diag[diag_idx + offset * (1ULL << (i + 2)) + (1ULL << (i + 1))] *=
718 std::exp(0.5 * i_ * PI * ucrzs[i][j]);
719 }
720 }
721 }
722 return std::make_pair(circ, diag);
723}

725void MultiplexedU2Box::generate_circuit() const {
726 Circuit circ;
727 Eigen::VectorXcd diag_vec;
728 std::tie(circ, diag_vec) = decompose();
729 if (impl_diag_ &&
730 (diag_vec - Eigen::VectorXcd::Constant(1ULL << circ.n_qubits(), 1))
731 .cwiseAbs()
732 .sum() > EPS) {
733 std::vector<unsigned> args(circ.n_qubits());
734 std::iota(std::begin(args), std::end(args), 0);
735 circ.add_box(DiagonalBox(diag_vec), args);
736 }
737 circ_ = std::make_shared<Circuit>(circ);

863void MultiplexedTensoredU2Box::generate_circuit() const {
864 Circuit circ(n_controls_ + n_targets_);
865 // contains the multiplexed-Rz gates
866 Circuit diag_circ(n_controls_ + n_targets_);
867 // the final diagonal on the control qubits
868 Eigen::VectorXcd diag_vec =
869 Eigen::VectorXcd::Constant(1ULL << n_controls_, 1);
870 for (unsigned i = 0; i < n_targets_; i++) {
871 ctrl_op_map_t u2_op_map;
872 for (auto it = op_map_.begin(); it != op_map_.end(); it++) {
873 u2_op_map.insert({it->first, it->second[i]});
874 }
875 MultiplexedU2Box mbox(u2_op_map);
876 Circuit inner_circ;
877 Eigen::VectorXcd inner_diag_vec;
878 std::tie(inner_circ, inner_diag_vec) = mbox.decompose();
879 std::vector<unsigned> args(n_controls_);
880 std::iota(std::begin(args), std::end(args), 0);
881 args.push_back(i + n_controls_);
882 // append the first part of the decomposition
883 circ.append_qubits(inner_circ, args);
884 // disentangle one qubit from the diagonal
885 // results in a multiplexed-Rz targeting the target j
886 ctrl_op_map_t multip_rz;
887 for (unsigned long long j = 0; j < (1ULL << n_controls_); j++) {
888 Complex a = inner_diag_vec[2 * j];
889 Complex b = inner_diag_vec[2 * j + 1];
890 // convert diag[a,b] into a p*Rz(alpha)
891 double a_phase = std::arg(a);
892 double b_phase = std::arg(b);
893 double alpha = (b_phase - a_phase) / PI;
894 Complex p = std::exp((b_phase + a_phase) * 0.5 * i_);
895 std::vector<bool> bitstr = dec_to_bin(j, n_controls_);
896 if (std::abs(alpha) > EPS) {
897 multip_rz.insert({bitstr, get_op_ptr(OpType::Rz, alpha)});
898 }
899 // update the diagonal on the control qubits
900 diag_vec[j] *= p;
901 }
902 if (!multip_rz.empty()) {
903 diag_circ.add_box(MultiplexedRotationBox(multip_rz), args);
904 }
905 }
906
907 circ.append(diag_circ);
908 if ((diag_vec - Eigen::VectorXcd::Constant(1ULL << n_controls_, 1))
909 .cwiseAbs()
910 .sum() > EPS) {
911 std::vector<unsigned> args(n_controls_);
912 std::iota(std::begin(args), std::end(args), 0);
913 circ.add_box(DiagonalBox(diag_vec), args);
914 }
915
916 circ_ = std::make_shared<Circuit>(circ);
917}
