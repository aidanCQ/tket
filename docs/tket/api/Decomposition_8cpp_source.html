107static bool convert_to_zyz(Circuit &circ) {
108 static const Expr half = SymEngine::div(Expr(1), Expr(2));
109 bool success = decompose_single_qubits_TK1().apply(circ);
110 VertexList bin;
111 BGL_FORALL_VERTICES(v, circ.dag, DAG) {
112 if (circ.n_in_edges(v) != 1) continue;
113 const Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
114 if (op->get_type() == OpType::TK1) {
115 std::vector<Expr> params = op->get_params();
116 Circuit replacement(1);
117 Expr a = params[2] + half;
118 Expr b = params[1];
119 Expr c = params[0] - half;
120 if (!equiv_0(a, 4)) {
121 replacement.add_op<unsigned>(OpType::Rz, a, {0});
122 }
123 if (!equiv_0(b, 4)) {
124 replacement.add_op<unsigned>(OpType::Ry, b, {0});
125 }
126 if (!equiv_0(c, 4)) {
127 replacement.add_op<unsigned>(OpType::Rz, c, {0});
128 }
129 Subcircuit sub = {
130 {circ.get_in_edges(v)}, {circ.get_all_out_edges(v)}, {v}};
131 bin.push_back(v);
132 circ.substitute(replacement, sub, Circuit::VertexDeletion::No);
133 success = true;
134 }
135 }
136 circ.remove_vertices(
137 bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
138 return success;
139}

202Transform decompose_ZYZ_to_TK1() {
203 return Transform([](Circuit &circ) {
204 bool success = false;
205 static const Expr zero(0);
206 static const Expr half = SymEngine::div(Expr(1), Expr(2));
207 VertexList bin;
208 VertexVec inputs = circ.q_inputs();
209 for (VertexVec::iterator i = inputs.begin(); i != inputs.end(); ++i) {
210 Edge e = circ.get_nth_out_edge(*i, 0);
211 Vertex v = circ.target(e);
212 while (!is_final_q_type(circ.get_OpType_from_Vertex(v))) {
213 if (circ.get_OpType_from_Vertex(v) == OpType::Rz) {
214 const Op_ptr v_g = circ.get_Op_ptr_from_Vertex(v);
215 Expr angle_1 = v_g->get_params()[0];
216 Edge e1 = circ.get_next_edge(v, e);
217 Vertex v2 = circ.target(e1);
218 if (circ.get_OpType_from_Vertex(v2) == OpType::Ry) {
219 const Op_ptr v2_g = circ.get_Op_ptr_from_Vertex(v2);
220 Expr angle_2 = v2_g->get_params()[0];
221 Edge e2 = circ.get_next_edge(v2, e1);
222 Vertex v3 = circ.target(e2);
223 bin.push_back(v2);
224 circ.remove_vertex(
225 v2, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
226 Expr angle_3 = zero;
227 if (circ.get_OpType_from_Vertex(v3) == OpType::Rz) {
228 const Op_ptr v3_g = circ.get_Op_ptr_from_Vertex(v3);
229 angle_3 = v3_g->get_params()[0];
230 circ.remove_vertex(
231 v3, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
232 bin.push_back(v3);
233 }
234 std::vector<Expr> new_params = {
235 angle_3 + half, angle_2, angle_1 - half};
236 circ.dag[v] = {get_op_ptr(OpType::TK1, new_params)};
237 } else {
238 circ.dag[v] = {get_op_ptr(OpType::TK1, {zero, zero, angle_1})};
239 }
240 } else if (circ.get_OpType_from_Vertex(v) == OpType::Ry) {
241 const Op_ptr v_g = circ.get_Op_ptr_from_Vertex(v);
242 Expr angle_2 = v_g->get_params()[0];
243 Expr angle_3 = zero;
244 Edge e1 = circ.get_next_edge(v, e);
245 Vertex v2 = circ.target(e1);
246 if (circ.get_OpType_from_Vertex(v2) == OpType::Rz) {
247 const Op_ptr v2_g = circ.get_Op_ptr_from_Vertex(v2);
248 angle_3 = v2_g->get_params()[0];
249 circ.remove_vertex(
250 v2, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
251 bin.push_back(v2);
252 }
253 std::vector<Expr> new_params = {angle_3 + half, angle_2, -half};
254 circ.dag[v] = {get_op_ptr(OpType::TK1, new_params)};
255 }
256 e = circ.get_next_edge(v, e);
257 v = circ.target(e);
258 }
259 }
260 circ.remove_vertices(
261 bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
262 return success;
263 });

266static Circuit tk1_angles_to_circ(Expr a, Expr b, Expr c) {
267 Circuit circ(1);
268
269 std::optional<double> ea = eval_expr(a);
270 std::optional<double> eb = eval_expr(b);
271 std::optional<double> ec = eval_expr(c);
272
273 // Remove global phases
274 if (ea && *ea >= 2) {
275 a -= 2;
276 circ.add_phase(1);
277 }
278 if (eb && *eb >= 2) {
279 b -= 2;
280 circ.add_phase(1);
281 }
282 if (ec && *ec >= 2) {
283 c -= 2;
284 circ.add_phase(1);
285 }
286
287 if (!equiv_0(a, 2) || !equiv_0(b, 2) || !equiv_0(c, 2)) {
288 circ.add_op<unsigned>(OpType::TK1, {c, b, a}, {0});
289 }
290
291 return circ;
292}

294Transform decompose_ZXZ_to_TK1() {
295 return Transform([](Circuit &circ) {
296 bool success = false;
297 VertexList bin;
298 VertexVec inputs = circ.q_inputs();
299 for (VertexVec::iterator i = inputs.begin(); i != inputs.end(); ++i) {
300 Edge e = circ.get_nth_out_edge(*i, 0);
301 Vertex v = circ.target(e);
302 // Angles for current TK1
303 std::array<Expr, 3> curr_angles;
304 // Index from 0 <= curr_ind <= 2 into `curr_angles` collecting TK1 angles
305 unsigned curr_ind = 0;
306 // The beginning edge of a sequence of Rx and Rz gates
307 std::optional<Edge> first_edge;
308 // Vertices of the Rx/Rz sequence. Empty iff first_edge == std::nullopt
309 VertexSet curr_vs;
310 while (true) {
311 Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
312 bool repeat_loop = false;
313 bool substitute_vertex = false;
314 TKET_ASSERT(curr_ind < 3);
315 switch (op->get_type()) {
316 case OpType::Rz: {
317 if (curr_ind == 1) {
318 curr_angles[curr_ind++] = 0;
319 }
320 TKET_ASSERT(op->get_params().size() == 1);
321 curr_angles[curr_ind++] = op->get_params().at(0);
322 substitute_vertex = true;
323 break;
324 }
325 case OpType::Rx: {
326 if (curr_ind != 1) {
327 curr_angles[curr_ind++] = 0;
328 }
329 TKET_ASSERT(op->get_params().size() == 1);
330 if (curr_ind < 3) {
331 curr_angles[curr_ind++] = op->get_params().at(0);
332 substitute_vertex = true;
333 } else {
334 repeat_loop = true;
335 }
336 break;
337 }
338 default: {
339 while (curr_ind < 3) {
340 curr_angles[curr_ind++] = 0;
341 }
342 }
343 }
344 if (substitute_vertex) {
345 curr_vs.insert(v);
346 if (!first_edge) {
347 first_edge = e;
348 }
349 }
350 // Substitute sequence of RzRxRz with TK1
351 auto substitute = [&]() {
352 if (first_edge) {
353 success = true;
354 Circuit sub = tk1_angles_to_circ(
355 curr_angles[0], curr_angles[1], curr_angles[2]);
356 Subcircuit hole{{*first_edge}, {e}, curr_vs};
357 // Backup
358 port_t backup_p = circ.get_target_port(e);
359 // Substitute
360 circ.substitute(sub, hole, Circuit::VertexDeletion::No);
361 // Restore
362 e = circ.get_nth_in_edge(v, backup_p);
363 // Reset all tracking variables
364 bin.insert(bin.end(), curr_vs.begin(), curr_vs.end());
365 std::fill(curr_angles.begin(), curr_angles.end(), 0);
366 curr_vs.clear();
367 first_edge.reset();
368 }
369 curr_ind = 0;
370 };
371 // Depending on `substitute_vertex`, we either place the TK1 before
372 // moving the edge forward or after
373 if (curr_ind == 3 && !substitute_vertex) {
374 substitute();
375 }
376 if (is_final_q_type(op->get_type())) {
377 TKET_ASSERT(!substitute_vertex);
378 break;
379 }
380 if (!repeat_loop) {
381 // Move edge forward
382 e = circ.get_next_edge(v, e);
383 v = circ.target(e);
384 }
385 if (curr_ind == 3 && substitute_vertex) {
386 substitute();
387 }
388 }
389 }
390 circ.remove_vertices(
391 bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
392 return success;
393 });
394}

459/* --Rz(a)--Rx(b)--R(c)-- => --Rz(a+c)--PhasedX(b,c)-- */
460Transform decompose_ZX_to_HQS1() {
461 return Transform([](Circuit &circ) {
462 bool success = false;
463 VertexList to_bin;
464 BGL_FORALL_VERTICES(v, circ.dag, DAG) {
465 if (circ.get_OpType_from_Vertex(v) == OpType::Rx) {
466 success = true;
467 const Op_ptr g = circ.get_Op_ptr_from_Vertex(v);
468 Expr theta = g->get_params()[0];
469 Vertex prev_vert = *circ.get_predecessors(v).begin();
470 Vertex next_vert = *circ.get_successors(v).begin();
471 if (circ.get_OpType_from_Vertex(prev_vert) == OpType::Rz &&
472 circ.get_OpType_from_Vertex(next_vert) == OpType::Rz) {
473 const Op_ptr prev_g = circ.get_Op_ptr_from_Vertex(prev_vert);
474 const Op_ptr next_g = circ.get_Op_ptr_from_Vertex(next_vert);
475 Expr phi = next_g->get_params()[0];
476 std::vector<Expr> params{theta, phi};
477 circ.dag[v].op = get_op_ptr(OpType::PhasedX, params);
478 circ.remove_vertex(
479 next_vert, Circuit::GraphRewiring::Yes,
480 Circuit::VertexDeletion::No);
481 to_bin.push_back(next_vert);
482 Expr new_param = prev_g->get_params()[0] + phi;
483 circ.dag[prev_vert].op = get_op_ptr(OpType::Rz, new_param);
484 } else {
485 // if no Rz, initialise a PhasedX op with theta=Rx.params[0],phi=0
486 Expr phi(0);
487 std::vector<Expr> params{theta, phi};
488 circ.dag[v].op = get_op_ptr(OpType::PhasedX, params);
489 }
490 }
491 }
492 circ.remove_vertices(
493 to_bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
494 remove_redundancies().apply(circ);
495 return success;
496 });
497}

499// Decompose CX into MolmerSorensen as:
500// ---C--- -V-S-|-H-
501// | --> XX(pi/4)
502// ---X--- -----|-Vdg-
503Transform decompose_MolmerSorensen() {
504 return Transform([](Circuit &circ) {
505 bool success = false;
506 VertexList bin;
507 BGL_FORALL_VERTICES(v, circ.dag, DAG) {
508 if (circ.get_OpType_from_Vertex(v) != OpType::CX) continue;
509 EdgeVec outs = circ.get_all_out_edges(v);
510 if (outs.size() == 2) {
511 Vertex next = circ.target(outs[0]);
512 // Is the next operation equivalent to an Rx, up to phase?
513 Op_ptr next_g = circ.get_Op_ptr_from_Vertex(next);
514 OpType next_type = next_g->get_type();
515 if (is_single_qubit_type(next_type) && !is_projective_type(next_type)) {
516 std::vector<Expr> angles = as_gate_ptr(next_g)->get_tk1_angles();
517 if (equiv_0(angles[0], 2) && equiv_0(angles[2], 2)) {
518 Expr angle = angles[1];
519 Expr phase = angles[3];
520 if (!equiv_0(angles[0], 4)) phase += 1;
521 if (!equiv_0(angles[2], 4)) phase += 1;
522 Edge next_e = circ.get_nth_out_edge(next, 0);
523 Vertex last = circ.target(next_e);
524 if (circ.get_OpType_from_Vertex(last) == OpType::CX &&
525 circ.get_nth_in_edge(last, 1) == outs[1]) {
526 // Recognise exp(-i XX * angle * pi/2)
527 const Op_ptr op_ptr = get_op_ptr(OpType::XXPhase, angle);
528 circ.dag[v] = {op_ptr};
529 bin.push_back(next);
530 circ.remove_vertex(
531 next, Circuit::GraphRewiring::Yes,
532 Circuit::VertexDeletion::No);
533 bin.push_back(last);
534 circ.remove_vertex(
535 last, Circuit::GraphRewiring::Yes,
536 Circuit::VertexDeletion::No);
537 circ.add_phase(phase);
538 success = true;
539 continue;
540 }
541 }
542 }
543 // Replace remaining CX gates
544 Subcircuit sub = {{circ.get_in_edges(v)}, {outs}, {v}};
545 bin.push_back(v);
546 circ.substitute(
547 CircPool::CX_using_XXPhase_1(), sub, Circuit::VertexDeletion::No);
548 success = true;
549 }
550 }
551 circ.remove_vertices(
552 bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
553 return success;
554 });
555}

578static double get_CX_fidelity(const std::array<double, 3> &k, unsigned nb_cx) {
579 TKET_ASSERT(nb_cx < 4);
580 auto [a, b, c] = k;
581 // gate fidelity achievable with 0,...,3 cnots
582 // this is fully determined by the information content k and is optimal
583 // see PhysRevA 71.062331 (2005) for more details on this
584 switch (nb_cx) {
585 case 0:
586 return trace_fidelity(a, b, c);
587 case 1:
588 return trace_fidelity(0.5 - a, b, c);
589 case 2:
590 return trace_fidelity(0, 0, c);
591 default:
592 return 1.;
593 }
594}

596// Try to decompose a TK2 gate using different gate sets, find the one with
597// the highest fidelity.
598// If no fidelities are provided, (best_optype, n_gates) is left unchanged.
599static double best_noise_aware_decomposition(
600 const std::array<double, 3> &angles, const TwoQbFidelities &fid,
601 OpType &best_optype, unsigned &n_gates) {
602 double max_fid = 0.;
603
604 // Try decomposition using CX or equivalent gates.
605 double cx_fid = std::max(
606 fid.CX_fidelity ? fid.CX_fidelity.value() : 0.,
607 fid.ZZMax_fidelity ? fid.ZZMax_fidelity.value() : 0.);
608 bool zzmax_is_better = false;
609 if (cx_fid < EPS) {
610 if (!fid.ZZPhase_fidelity) {
611 // No fidelity is defined, so default to CX
612 cx_fid = 1.;
613 }
614 } else {
615 zzmax_is_better = fid.CX_fidelity < fid.ZZMax_fidelity;
616 }
617 if (cx_fid > EPS) {
618 for (unsigned n_cx = 0; n_cx <= 3; ++n_cx) {
619 double ncx_fid = get_CX_fidelity(angles, n_cx) * pow(cx_fid, n_cx);
620 if (ncx_fid > max_fid) {
621 max_fid = ncx_fid;
622 best_optype = zzmax_is_better ? OpType::ZZMax : OpType::CX;
623 n_gates = n_cx;
624 }
625 }
626 }
627
628 // Try decomposition using ZZPhase(α)
629 if (fid.ZZPhase_fidelity) {
630 double zz_fid = 1.;
631 // If ZZMax is available, ZZPhase is only interesting when used once.
632 // (two ZZPhase can always be written using two ZZmax)
633 unsigned max_nzz = fid.ZZMax_fidelity ? 1 : 3;
634 for (unsigned n_zz = 0; n_zz <= max_nzz; ++n_zz) {
635 if (n_zz > 0) {
636 double gate_fid = (*fid.ZZPhase_fidelity)(angles[n_zz - 1]);
637 if (gate_fid < 0 || gate_fid > 1) {
638 throw std::domain_error(
639 "ZZPhase_fidelity returned a value outside of [0, 1].");
640 }
641 zz_fid *= gate_fid;
642 }
643 double nzz_fid = get_ZZPhase_fidelity(angles, n_zz) * zz_fid;
644 // Use ZZPhase if fidelity is greater or it is equal but uses fewer gates
645 if (nzz_fid - max_fid > EPS ||
646 (nzz_fid - max_fid > -EPS && n_zz < n_gates)) {
647 max_fid = nzz_fid;
648 best_optype = OpType::ZZPhase;
649 n_gates = n_zz;
650 }
651 }
652 }
653
654 return max_fid;
655}

657// Try to decompose a TK2 gate exactly using different gate sets.
658// The fidelities are used as an indication of which gate set is usable, but
659// the actual values of the fidelities will be ignored.
660//
661// Relies on default values of best_optype and n_gates if no optimisation can be
662// performed.
663static void best_exact_decomposition(
664 const std::array<Expr, 3> &angles, const TwoQbFidelities &fid,
665 OpType &best_optype, unsigned &n_gates) {
666 // Prefer CX/ZZMax when possible.
667 if (fid.CX_fidelity || fid.ZZMax_fidelity) {
668 bool zzmax_is_better =
669 !fid.CX_fidelity || fid.CX_fidelity < fid.ZZMax_fidelity;
670 best_optype = zzmax_is_better ? OpType::ZZMax : OpType::CX;
671 } else if (fid.ZZPhase_fidelity) {
672 // Only ZZPhase has fidelity, so use ZZPhase.
673 best_optype = OpType::ZZPhase;
674 }
675
676 if (best_optype == OpType::CX || best_optype == OpType::ZZMax) {
677 // Reduce n_gates if possible.
678 if (equiv_0(angles[2], 4)) {
679 n_gates = 2;
680 }
681 } else if (best_optype == OpType::ZZPhase) {
682 // Reduce n_gates if possible.
683 if (equiv_0(angles[2], 4)) {
684 n_gates = 2;
685 if (equiv_0(angles[1], 4)) {
686 n_gates = 1;
687 }
688 }
689 }
690
691 // Finally, handle the only case where ZZPhase is preferable over ZZMax.
692 if (fid.ZZPhase_fidelity && equiv_0(angles[2], 4) && equiv_0(angles[1], 4) &&
693 n_gates > 1) {
694 n_gates = 1;
695 best_optype = OpType::ZZPhase;
696 }
697}

717static Circuit TK2_replacement(
718 std::array<Expr, 3> angles, const TwoQbFidelities &fid, bool allow_swaps) {
719 if (!in_weyl_chamber(angles)) {
720 throw std::domain_error("TK2 params are not normalised to Weyl chamber.");
721 }
722 OpType best_optype = OpType::CX; // default to using CX
723 unsigned n_gates = 3; // default to 3x CX
724 bool implicit_swap = false; // default to no implicit swap
725
726 // Only used when allow_swaps == true.
727 Circuit pre, post;
728 std::array<Expr, 3> angles_swapped;
729 if (allow_swaps) {
730 // Swapped circuit
731 Circuit swap_circ(2);
732 angles_swapped = angles;
733 for (unsigned i = 0; i < 3; ++i) {
734 angles_swapped[i] += 0.5;
735 }
736 std::tie(pre, angles_swapped, post) = normalise_TK2_angles(
737 angles_swapped[0], angles_swapped[1], angles_swapped[2]);
738 pre.add_phase(0.25);
739 }
740
741 // Try to evaluate exprs to doubles.
742 std::array<double, 3> angles_eval;
743 std::array<double, 3> angles_eval_swapped;
744 unsigned last_angle = 0;
745 for (; last_angle < 3; ++last_angle) {
746 std::optional<double> eval = eval_expr_mod(angles[last_angle]);
747 if (eval) {
748 angles_eval[last_angle] = *eval;
749 } else {
750 break;
751 }
752 if (allow_swaps) {
753 eval = eval_expr_mod(angles_swapped[last_angle]);
754 TKET_ASSERT(eval);
755 angles_eval_swapped[last_angle] = *eval;
756 }
757 }
758
759 if (last_angle <= 2) {
760 // Not all angles could be resolved numerically.
761 // For symbolic angles, we can only provide an exact decomposition.
762 best_exact_decomposition(angles, fid, best_optype, n_gates);
763 if (allow_swaps) {
764 OpType best_optype_swapped = OpType::CX; // default to using CX
765 unsigned n_gates_swapped = 3; // default to 3x CX
766 best_exact_decomposition(
767 angles_swapped, fid, best_optype_swapped, n_gates_swapped);
768 if (n_gates_swapped < n_gates) {
769 n_gates = n_gates_swapped;
770 best_optype = best_optype_swapped;
771 angles = angles_swapped;
772 implicit_swap = true;
773 }
774 }
775 } else {
776 // For non-symbolic angles, we can find the optimal number of gates
777 // using the gate fidelities provided.
778 double max_fid =
779 best_noise_aware_decomposition(angles_eval, fid, best_optype, n_gates);
780 if (allow_swaps) {
781 OpType best_optype_swapped = OpType::CX; // default to using CX
782 unsigned n_gates_swapped = 3; // default to 3x CX
783 double max_fid_swapped = best_noise_aware_decomposition(
784 angles_eval_swapped, fid, best_optype_swapped, n_gates_swapped);
785 if (max_fid_swapped > max_fid ||
786 ((max_fid_swapped - max_fid) < EPS && n_gates_swapped < n_gates)) {
787 n_gates = n_gates_swapped;
788 best_optype = best_optype_swapped;
789 angles = angles_swapped;
790 implicit_swap = true;
791 }
792 }
793 }
794
795 // Build circuit for substitution.
796 Circuit sub(2);
797 switch (best_optype) {
798 case OpType::ZZMax:
799 case OpType::CX: {
800 switch (n_gates) {
801 case 0:
802 break;
803 case 1: {
804 sub.append(CircPool::approx_TK2_using_1xCX());
805 break;
806 }
807 case 2: {
808 sub.append(CircPool::approx_TK2_using_2xCX(angles[0], angles[1]));
809 break;
810 }
811 case 3: {
812 sub.append(CircPool::TK2_using_3xCX(angles[0], angles[1], angles[2]));
813 break;
814 }
815 default:
816 TKET_ASSERT(!"Number of CX invalid in decompose_TK2");
817 }
818 if (best_optype == OpType::ZZMax) {
819 decompose_CX_to_HQS2().apply(sub);
820 }
821 break;
822 }
823 case OpType::ZZPhase: {
824 switch (n_gates) {
825 case 0:
826 break;
827 case 1: {
828 sub.append(CircPool::approx_TK2_using_1xZZPhase(angles[0]));
829 break;
830 }
831 case 2: {
832 sub.append(
833 CircPool::approx_TK2_using_2xZZPhase(angles[0], angles[1]));
834 break;
835 }
836 case 3: {
837 sub.append(
838 CircPool::TK2_using_ZZPhase(angles[0], angles[1], angles[2]));
839 break;
840 }
841 default:
842 TKET_ASSERT(!"Number of ZZPhase invalid in decompose_TK2");
843 }
844 break;
845 }
846 default:
847 throw BadOpType(
848 "Unrecognised target OpType in decompose_TK2", best_optype);
849 }
850
851 if (implicit_swap) {
852 Circuit swap(2);
853 swap.add_op<unsigned>(OpType::SWAP, {0, 1});
854 sub = pre >> sub >> post >> swap;
855 sub.replace_SWAPs();
856 }
857
858 return sub;
859}

866 if (fid.ZZMax_fidelity) {
867 if (*fid.ZZMax_fidelity < 0 || *fid.ZZMax_fidelity > 1) {
868 throw std::domain_error("ZZMax fidelity must be between 0 and 1.");
869 }
870 }
871 if (fid.CX_fidelity) {

873 throw std::domain_error("CX fidelity must be between 0 and 1.");
874 }
875 }
876 if (fid.ZZMax_fidelity && fid.ZZPhase_fidelity) {
877 if (*fid.ZZMax_fidelity < (*fid.ZZPhase_fidelity)(.5)) {

879 "The ZZMax fidelity cannot be smaller than the ZZPhase(0.5) "
880 "fidelity");

884 bool success = false;
885

889
890 success = true;

892 TKET_ASSERT(params.size() == 3);
893 std::array<Expr, 3> angles{params[0], params[1], params[2]};

895 Circuit sub = TK2_replacement(angles, fid, allow_swaps);
896 bin.push_back(v);
897 circ.substitute(sub, v, Circuit::VertexDeletion::No);
898 }
899 circ.remove_vertices(
900 bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
901
902 return success;
903 });

1116 type != OpType::Z && is_single_qubit_unitary_type(type)) {

1250Transform decompose_PhaseGadgets() {
1251 return Transform([](Circuit &circ) {
1252 bool success = false;
1253 VertexList big_bin;
1254 auto [it, end] = boost::vertices(circ.dag);
1255 for (auto next = it; it != end; it = next) {
1256 ++next;
1257 if (circ.get_OpType_from_Vertex(*it) == OpType::CX &&
1258 circ.n_out_edges(*it) == 2) {
1259 EdgeVec outs = circ.get_all_out_edges(*it);
1260 Vertex next_v = circ.target(outs[1]);
1261 Op_ptr g = circ.get_Op_ptr_from_Vertex(next_v);
1262 OpType type = g->get_type();
1263 if (type == OpType::Rz || type == OpType::U1 ||
1264 (type == OpType::TK1 && equiv_0(g->get_params()[1]))) {
1265 Vertex last_v = circ.get_next_pair(next_v, outs[1]).first;
1266 if (circ.get_OpType_from_Vertex(last_v) == OpType::CX &&
1267 circ.get_nth_in_edge(last_v, 0) == outs[0]) {
1268 VertexList bin = {next_v, last_v};
1269 big_bin.push_back(next_v);
1270 big_bin.push_back(last_v);
1271 circ.remove_vertices(
1272 bin, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
1273 Expr t = g->get_params()[0];
1274 if (type == OpType::TK1) {
1275 t += g->get_params()[2];
1276 }
1277 circ.dag[*it] = {get_op_ptr(OpType::PhaseGadget, {t}, 2)};
1278 if (type == OpType::U1) {
1279 circ.add_phase(t / 2);
1280 } else if (

1370Transform decompose_SWAP_to_CX(const Architecture &arc) {
1371 // Note that the default argument will be out of scope at call-time!
1372 // => we replace the default empty Architecture with nullptr
1373 // we need to keep arc as a pointer as there is no such thing as
1374 // optional references in std
1375 const Architecture *arc_ptr = arc.n_nodes() ? &arc : nullptr;
1376 return Transform([arc_ptr](Circuit &circ) {
1377 bool success = false;
1378 std::vector<std::pair<Vertex, bool>> bin;
1379 for (Circuit::CommandIterator it = circ.begin(); it != circ.end(); ++it) {
1380 if (it->get_op_ptr()->get_type() == OpType::SWAP) {
1381 unit_vector_t qbs = it->get_args();
1382 node_vector_t nodes = {qbs.begin(), qbs.end()};
1383 if (arc_ptr != nullptr && arc_ptr->node_exists(nodes[0]) &&
1384 arc_ptr->node_exists(nodes[1]) &&
1385 arc_ptr->edge_exists(nodes[1], nodes[0])) {
1386 bin.push_back({it.get_vertex(), true});
1387 } else {
1388 bin.push_back({it.get_vertex(), false});
1389 }
1390 }
1391 }
1392
1393 for (std::pair<Vertex, bool> v : bin) {
1394 success = true;
1395 // Get predecessor vertices and successor vertices and find subcircuit
1396 // for replacement
1397 VertexVec preds = circ.get_predecessors(v.first);
1398 VertexVec succs = circ.get_successors(v.first);
1399 EdgeVec in_edges = circ.get_in_edges(v.first);
1400 EdgeVec out_edges = circ.get_all_out_edges(v.first);
1401 Subcircuit sub = {in_edges, out_edges, {v.first}};
1402
1403 if (preds.size() == 1 &&
1404 circ.get_OpType_from_Vertex(preds[0]) == OpType::CX) {
1405 // if conditions requires that there is a CX gate before SWAP
1406 swap_sub(
1407 circ, CircPool::SWAP_using_CX_0(), CircPool::SWAP_using_CX_1(), sub,
1408 {circ.get_source_port(in_edges[0]),
1409 circ.get_source_port(in_edges[1])});
1410 } else if (
1411 succs.size() == 1 &&
1412 circ.get_OpType_from_Vertex(succs[0]) == OpType::CX) {
1413 // if no CX gate before SWAP, check after.
1414 swap_sub(
1415 circ, CircPool::SWAP_using_CX_0(), CircPool::SWAP_using_CX_1(), sub,
1416 {circ.get_target_port(out_edges[0]),
1417 circ.get_target_port(out_edges[1])});
1418 } else if (v.second) {
1419 // We assume in general that a CX gate saving is desired over H gate
1420 // savings. If SWAP doesn't lend itself to annihlation though, the
1421 // SWAP is inserted to reduce number of H gates added in a 'directed'
1422 // CX decomposition. SWAP_using_CX_1 is added if the backwards
1423 // direction is available on the architecture
1424 circ.substitute(
1425 CircPool::SWAP_using_CX_1(), sub, Circuit::VertexDeletion::Yes);
1426 } else {
1427 // SWAP_using_CX_0 is added as a default option
1428 circ.substitute(
1429 CircPool::SWAP_using_CX_0(), sub, Circuit::VertexDeletion::Yes);
1430 }
1431 }
1432 return success;
1433 });

1436Transform decompose_BRIDGE_to_CX() {
1437 return Transform([](Circuit &circ) {
1438 bool success = false;
1439 // Collect all BRIDGE type vertices
1440 std::vector<std::pair<Vertex, bool>> bin;
1441 BGL_FORALL_VERTICES(v, circ.dag, DAG) {
1442 if (circ.get_OpType_from_Vertex(v) == OpType::BRIDGE) {
1443 bin.push_back({v, false});
1444 }
1445 if (circ.get_OpType_from_Vertex(v) == OpType::Conditional) {
1446 const Conditional &b =
1447 static_cast<const Conditional &>(*circ.get_Op_ptr_from_Vertex(v));
1448 if (b.get_op()->get_type() == OpType::BRIDGE) {
1449 bin.push_back({v, true});
1450 }
1451 }
1452 }
1453
1454 auto BRIDGE_sub =
1455 [&circ](std::pair<Vertex, bool> &candidate, Circuit BRIDGE_circ) {
1456 if (candidate.second)
1457 circ.substitute_conditional(
1458 BRIDGE_circ, candidate.first, Circuit::VertexDeletion::Yes);
1459 else
1460 circ.substitute(
1461 BRIDGE_circ, candidate.first, Circuit::VertexDeletion::Yes);
1462 };
1463
1464 for (std::pair<Vertex, bool> v : bin) {
1465 // Get predecessor vertices and successor vertices and find subcircuit
1466 // for replacement
1467 success = true;
1468 VertexVec preds = circ.get_predecessors(v.first);
1469 VertexVec succs = circ.get_successors(v.first);
1470 EdgeVec in_edges = circ.get_in_edges(v.first);
1471 EdgeVec out_edges = circ.get_all_out_edges(v.first);
1472 Subcircuit sub = {in_edges, out_edges, {v.first}};
1473
1474 bool done = false;
1475 if (preds.size() < 3) { // Implies some predecessor vertices are in a
1476 // multi-qubit op together
1477 VertexVec comps = {
1478 circ.source(in_edges[0]), circ.source(in_edges[1]),
1479 circ.source(in_edges[2])};
1480 if (comps[0] ==
1481 comps[1]) { // First two qubits in BRIDGE in multi-qubit op
1482 BRIDGE_sub(v, CircPool::BRIDGE_using_CX_0());
1483 done = true;
1484 } else if (comps[2] == comps[1]) { // Second two qubits in BRIDGE in
1485 // multi-qubit op together before
1486 // BRIDGE
1487 BRIDGE_sub(v, CircPool::BRIDGE_using_CX_1());
1488 done = true;
1489 }
1490 }
1491 if (succs.size() < 3 && !done) { // Implies some successor vertices are
1492 // in a multiq-ubit op together
1493 VertexVec comps = {
1494 circ.target(out_edges[0]), circ.target(out_edges[1]),
1495 circ.target(out_edges[2])};
1496 if (comps[0] == comps[1]) { // First two qubits in BRIDGE in
1497 // multi-qubit op together before BRIDGE
1498 BRIDGE_sub(v, CircPool::BRIDGE_using_CX_1());
1499 done = true;
1500 } else if (comps[2] == comps[1]) { // Second two qubits in BRIDGE in
1501 // multi-qubit op together before
1502 // BRIDGE
1503 BRIDGE_sub(v, CircPool::BRIDGE_using_CX_0());
1504 done = true;
1505 }
1506 }
1507 if (!done) { // default decomposition
1508 BRIDGE_sub(v, CircPool::BRIDGE_using_CX_1());
1509 }
1510 }
1511 return success;
1512 });

1609static unsigned n_distinct_beta(const Circuit &circ, const OptVertexVec &gates);

1611static bool all_equal(const std::vector<T> &vs);
1612
1613// Any PhasedX or NPhasedX gate is replaced by an NPhasedX that is global
1614Transform globalise_PhasedX(bool squash) {
1615 // The key bit: choose the decomposition strategy depending on the current
1616 // beta angles.
1617 //
1618 // Given the set of PhasedX gates to synthesise, choose which decomposition
1619 // strategy. Valid strategies are 0, 1, 2, corresponding to the number of
1620 // NPhasedX gates to be inserted.
1621 //
1622 // The current strategy is rather simple: it chooses to insert a single
1623 // NPhasedX whenever it would solve the current problem and there are further
1624 // PhasedX left (meaning that the rest of the computation can be deferred till
1625 // later), otherwise inserts 2x PhasedX.
1626 auto choose_strategy = [](const PhasedXFrontier &frontier, unsigned n_target,
1627 unsigned n_all) {
1628 if (n_all == 0 || n_target == 0) {
1629 return 0;
1630 }
1631 if (n_target == 1 && n_all == 1) {
1632 return 1;
1633 }
1634 if (n_target == 1 && frontier.are_phasedx_left()) {
1635 return 1;
1636 }
1637 return 2;
1638 };
1639
1640 // the actual transform
1641 return Transform([squash, choose_strategy](Circuit &circ) {
1642 // if we squash, we start by removing all NPhasedX gates
1643 if (squash) {
1644 Transforms::decompose_NPhasedX().apply(circ);
1645 }
1646
1647 std::vector<unsigned> range_qbs(circ.n_qubits());
1648 std::iota(range_qbs.begin(), range_qbs.end(), 0);
1649 PhasedXFrontier frontier(circ);
1650
1651 // find a total ordering of boundary gates (aka non-NPhasedX multi-qb gates)
1652 auto is_boundary = [&circ](Vertex v) {
1653 Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
1654 return PhasedXFrontier::is_interval_boundary(op);
1655 };
1656 // get a lexicographic ordering of boundary vertices
1657 auto vertices_in_order = circ.vertices_in_order();
1658 auto r = vertices_in_order | boost::adaptors::filtered(is_boundary);
1659 OptVertexVec boundary_gates(r.begin(), r.end());
1660 // Add sentinel to process the gates after last boundary_gate.
1661 // This is required to force flush the frontier after the last multi-qubit
1662 // gate.
1663 OptVertex optv;
1664 boundary_gates.push_back(optv);
1665
1666 // whether transform is successful (always true if squash=true)
1667 bool success = squash;
1668
1669 // Loop through each multi-qb gate.
1670 // At each iteration, decide how to decompose the single-qb gates
1671 // immediately preceding the current multi-qb gate into global gates.
1672 //
1673 // The rationale: defer any computation until just before the next multi-qb
1674 // gate. At that point, the single-qb gates on the qubits where the multi-qb
1675 // gate is acting HAVE TO be decomposed. Figure out how to do that and take
1676 // care of the garbage you might have created on other qubits at a later
1677 // point.
1678
1679 for (OptVertex v : boundary_gates) {
1680 // the qubits whose intervals must be decomposed into global gates
1681 std::set<unsigned> curr_qubits;
1682 while (true) {
1683 if (v) {
1684 curr_qubits = frontier.qubits_ending_in(*v);
1685 } else {
1686 curr_qubits.clear();
1687 for (unsigned i = 0; i < circ.n_qubits(); ++i) {
1688 curr_qubits.insert(curr_qubits.end(), i);
1689 }
1690 }
1691 if (squash) {
1692 frontier.squash_intervals();
1693 }
1694 OptVertexVec all_phasedx = frontier.get_all_beta_vertices();
1695 OptVertexVec curr_phasedx;
1696 for (unsigned q : curr_qubits) {
1697 curr_phasedx.push_back(all_phasedx[q]);
1698 }
1699
1700 if (all_nullopt(curr_phasedx)) {
1701 // there is nothing to decompose anymore, move to next boundary_gate
1702 break;
1703 }
1704 if (all_equal(all_phasedx)) {
1705 // this is already a global NPhasedX gate, leave untouched
1706 frontier.skip_global_gates(1);
1707 continue;
1708 }
1709
1710 // find best decomposition strategy
1711 unsigned n_curr_betas = n_distinct_beta(circ, curr_phasedx);
1712 unsigned n_all_betas = n_distinct_beta(circ, all_phasedx);
1713 unsigned strategy;
1714 if (squash) {
1715 strategy = choose_strategy(frontier, n_curr_betas, n_all_betas);
1716 } else {
1717 // if we don't squash we decompose each NPhasedX with 2x global
1718 strategy = 2;
1719 }
1720 switch (strategy) {
1721 case 0:
1722 // do nothing
1723 break;
1724 case 1:
1725 // insert one single global NPhasedX
1726 TKET_ASSERT(curr_qubits.size() > 0);
1727 frontier.insert_1_phasedx(*(curr_qubits.begin()));
1728 success = true;
1729 break;
1730 case 2:
1731 // insert two global NPhasedX
1732 frontier.insert_2_phasedx();
1733 success = true;
1734 break;
1735 default:
1736 TKET_ASSERT(!"Invalid strategy in replace_non_global_phasedx");
1737 }
1738 }
1739 if (v) {
1740 frontier.next_multiqb(*v);
1741 }
1742 }
1743 TKET_ASSERT(frontier.is_finished());
1744
1745 success |= absorb_Rz_NPhasedX().apply(circ);
1746
1747 return success;
1748 });
1749}

1751// The number of distinct beta angles in `gates`.
1752//
1753// Performs O(n^2) pairwise equivalence comparisons between expressions to
1754// handle symbolic variables and floating point errors
1755unsigned n_distinct_beta(const Circuit &circ, const OptVertexVec &gates) {
1756 std::vector<Expr> vals;
1757
1758 // collect all epxressions
1759 for (OptVertex v : gates) {
1760 if (v) {
1761 Expr angle = circ.get_Op_ptr_from_Vertex(*v)->get_params()[0];
1762 vals.push_back(angle);
1763 } else {
1764 vals.push_back(0.);
1765 }
1766 }
1767
1768 unsigned n_distinct = 0;
1769
1770 // perform pairwise equivalence checks
1771 for (unsigned i = 0; i < vals.size(); ++i) {
1772 bool is_unique = true;
1773 for (unsigned j = i + 1; j < vals.size(); ++j) {
1774 if (equiv_expr(vals[i], vals[j])) {
1775 is_unique = false;
1776 break;
1777 }
1778 }
1779 n_distinct += is_unique;
1780 }
1781 return n_distinct;
1782}

1844static void substitute_cnx(
1845 const Command &cmd, const std::optional<qubit_vector_t> &new_args,
1846 const bool &as_dagger, Circuit &circ,
1847 std::map<unsigned, Circuit> &cnx_cache) {
1848 Circuit cnx;
1849 qubit_vector_t original_args = cmd.get_qubits();
1850 bool conditional = cmd.get_op_ptr()->get_type() == OpType::Conditional;
1851
1852 // Compute the decomposition
1853 auto cache = cnx_cache.find((unsigned)original_args.size());
1854 if (cache != cnx_cache.end()) {
1855 cnx = cache->second;
1856 } else {
1857 cnx = multi_controlled_to_2q(
1858 get_op_ptr(OpType::CnX, std::vector<Expr>(), original_args.size()));
1859 cnx_cache.insert({(unsigned)original_args.size(), cnx});
1860 }

1862 // Reorder the replacement circuit by renaming the qubits
1863 if (new_args != std::nullopt) {
1864 // The cnx circuit must only use default registers for this renaming to
1865 // work.
1866 TKET_ASSERT(cnx.is_simple());

1869 auto it =
1870 find(original_args.begin(), original_args.end(), new_args.value()[i]);
1871 int index = it - original_args.begin();
1872 qmap.insert({Qubit(i), Qubit(index)});
1873 }
1874 cnx.rename_units(qmap);
1875 }
1876
1877 if (as_dagger) {
1878 cnx = cnx.dagger();
1879 }
1880 if (conditional) {
1881 circ.substitute_conditional(
1882 cnx, cmd.get_vertex(), Circuit::VertexDeletion::No);
1883 } else {
1884 circ.substitute(cnx, cmd.get_vertex(), Circuit::VertexDeletion::No);
1885 }
1886}

1888Transform cnx_pairwise_decomposition() {
1889 return Transform([](Circuit &circ) {
1890 bool success = commute_through_multis().apply(circ);
1891 success |= remove_redundancies().apply(circ);
1892
1893 // Cache CnX decompositions
1894 std::map<unsigned, Circuit> cnx_cache;
1895 // Replaced vertices to delete at the end
1896 VertexList bin;
1897
1898 // Find all CnX gates, including conditional ones
1899 std::vector<Command> commands = circ.get_commands();
1900 std::vector<int> cnx_indices;
1901 for (unsigned i = 0; i < commands.size(); i++) {
1902 Op_ptr op = commands[i].get_op_ptr();
1903 bool conditional = op->get_type() == OpType::Conditional;
1904 if (conditional) {
1905 const Conditional &cond = static_cast<const Conditional &>(*op);
1906 op = cond.get_op();
1907 }
1908
1909 if (op->get_type() == OpType::CnX || op->get_type() == OpType::CCX) {
1910 if (op->n_qubits() < 3) {
1911 // Decompose as CX or X without further optimisation
1912 Circuit replacement = multi_controlled_to_2q(op);
1913 if (conditional) {
1914 circ.substitute_conditional(
1915 replacement, commands[i].get_vertex(),
1916 Circuit::VertexDeletion::No);
1917 } else {
1918 circ.substitute(
1919 replacement, commands[i].get_vertex(),
1920 Circuit::VertexDeletion::No);
1921 }
1922 bin.push_back(commands[i].get_vertex());
1923 } else {
1924 cnx_indices.push_back(i);
1925 }
1926 }
1927 }
1928
1929 // If the number of CnX is odd, decompose the first one as it is
1930 unsigned itr = 0;
1931 if (cnx_indices.size() % 2 == 1) {
1932 substitute_cnx(
1933 commands[cnx_indices[0]], std::nullopt, false, circ, cnx_cache);
1934 bin.push_back(commands[cnx_indices[0]].get_vertex());
1935 itr = 1;
1936 }
1937
1938 // For the rest of CnX, we reorder the control qubits of each pair
1939 // to maximise the chance of gate cancelling.
1940 for (; itr + 1 < cnx_indices.size(); itr += 2) {
1941 Command &cmd1 = commands[cnx_indices[itr]];
1942 Command &cmd2 = commands[cnx_indices[itr + 1]];
1943
1944 qubit_vector_t args1 = cmd1.get_qubits();
1945 qubit_vector_t args2 = cmd2.get_qubits();
1946
1947 qubit_vector_t new_args1 = args1;
1948 qubit_vector_t new_args2 = args2;
1949
1950 bool cmd1_as_dagger = false;
1951 bool cmd2_as_dagger = false;
1952
1953 if (args1.size() == 3 && args2.size() == 3) {
1954 // CCX gates are handled differently, there
1955 // is a maximum of 1 CX reduction (needs to be realised with
1956 // CliffordSimp), when they have args: [a,b,c] and [c,a,b]
1957 std::set<Qubit> args1_set(args1.begin(), args1.end());
1958 std::set<Qubit> args2_set(args2.begin(), args2.end());
1959 // Check if they are acting on the same set of qubits
1960 if (args1_set == args2_set) {
1961 // If they have the same target, we decompose them as CCX.dagger and
1962 // CCX
1963 if (new_args1.back() == new_args2.back()) {
1964 cmd1_as_dagger = true;
1965 new_args2 = new_args1;
1966 } else {
1967 // We find qubits a,b,c
1968 // such that the first CCX acts on [a,b,c]
1969 // and the second CCX acts on [c,a,b] without changing the semantics
1970 Qubit a = (args1[0] == args2.back()) ? args1[1] : args1[0];
1971 new_args1 = {a, args2.back(), args1.back()};
1972 new_args2 = {args1.back(), a, args2.back()};
1973 }
1974 }
1975 // If the two CCX gates are not acting on the same set of qubits
1976 // we don't change ther arguments, and decompose them as they are.
1977 } else {
1978 // The more general case
1979 // we try to reorder the control args of the two gates to maximise
1980 // cancellation. The idea is to move the common control qubits to the
1981 // end of the control qubit lists e.g. C4X [q0, q1, q2, q3, q4] and C4X
1982 // [q1, q2, q4, q7, q6] will be reordered as C4X [q0, q3, q1, q2, q4]
1983 // and C4X [q4, q7, q1, q2, q6]
1984 qubit_vector_t common_ctrls;
1985 qubit_vector_t ctrl_args1 = args1;
1986 qubit_vector_t ctrl_args2 = args2;
1987 Qubit target1 = ctrl_args1.back();
1988 Qubit target2 = ctrl_args2.back();
1989 // Remove the target qubits so they only have control qubits
1990 ctrl_args1.pop_back();
1991 ctrl_args2.pop_back();
1992 // common_ctrls = ctrl_args1 intersect ctrl_args2
1993 std::sort(ctrl_args1.begin(), ctrl_args1.end());
1994 std::sort(ctrl_args2.begin(), ctrl_args2.end());
1995 std::set_intersection(
1996 ctrl_args1.begin(), ctrl_args1.end(), ctrl_args2.begin(),
1997 ctrl_args2.end(), std::back_inserter(common_ctrls));
1998
1999 // reorder control qubits such that
2000 // new_args1 = (ctrl_args1 - common_ctrls) + common_ctrls + target1
2001 // new_args2 = (ctrl_args2 - common_ctrls) + common_ctrls + target2
2002 new_args1.clear();
2003 new_args2.clear();
2004 std::set_difference(
2005 ctrl_args1.begin(), ctrl_args1.end(), common_ctrls.begin(),
2006 common_ctrls.end(), std::back_inserter(new_args1));
2007 std::set_difference(
2008 ctrl_args2.begin(), ctrl_args2.end(), common_ctrls.begin(),
2009 common_ctrls.end(), std::back_inserter(new_args2));
2010 new_args1.insert(
2011 new_args1.end(), common_ctrls.begin(), common_ctrls.end());
2012 new_args2.insert(
2013 new_args2.end(), common_ctrls.begin(), common_ctrls.end());
2014 new_args1.push_back(target1);
2015 new_args2.push_back(target2);
2016 cmd2_as_dagger = true;
2017 }
2018 substitute_cnx(cmd1, new_args1, cmd1_as_dagger, circ, cnx_cache);
2019 substitute_cnx(cmd2, new_args2, cmd2_as_dagger, circ, cnx_cache);
2020 bin.push_back(cmd1.get_vertex());
2021 bin.push_back(cmd2.get_vertex());
2022 }
2023
2024 circ.remove_vertices(
2025 bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
2026 success |= remove_redundancies().apply(circ);
2027 success |= !bin.empty();
2028 return success;
2029 });
2030}
