49PassPtr gen_rebase_pass(
50 const OpTypeSet& allowed_gates, const Circuit& cx_replacement,
51 const std::function<Circuit(const Expr&, const Expr&, const Expr&)>&
52 tk1_replacement) {
53 Transform t = Transforms::rebase_factory(
54 allowed_gates, cx_replacement, tk1_replacement);
55
56 PredicatePtrMap precons;
57 OpTypeSet all_types(allowed_gates);
58 all_types.insert(OpType::Measure);
59 all_types.insert(OpType::Collapse);
60 all_types.insert(OpType::Reset);
61 PredicatePtr postcon1 = std::make_shared<GateSetPredicate>(all_types);
62 PredicatePtr postcon2 = std::make_shared<MaxTwoQubitGatesPredicate>();
63 std::pair<const std::type_index, PredicatePtr> pair2 =
64 CompilationUnit::make_type_pair(postcon1);
65 PredicatePtrMap s_postcons{pair2, CompilationUnit::make_type_pair(postcon2)};
66 PredicateClassGuarantees g_postcons{{pair2.first, Guarantee::Clear}};
67 PostConditions pc = {s_postcons, g_postcons, Guarantee::Preserve};
68 // record pass config
69 nlohmann::json j;
70 j["name"] = "RebaseCustom";
71 j["basis_allowed"] = allowed_gates;
72 j["basis_cx_replacement"] = cx_replacement;
73 j["basis_tk1_replacement"] =
74 "SERIALIZATION OF FUNCTIONS IS NOT YET SUPPORTED";
75 return std::make_shared<StandardPass>(precons, t, pc, j);
76}

195PassPtr gen_flatten_relabel_registers_pass(const std::string& label) {
196 Transform t =
197 Transform([=](Circuit& circuit, std::shared_ptr<unit_bimaps_t> maps) {
198 unsigned n_qubits = circuit.n_qubits();
199 circuit.remove_blank_wires();
200 bool changed = circuit.n_qubits() < n_qubits;
201 std::map<Qubit, Qubit> relabelling_map;
202 std::vector<Qubit> all_qubits = circuit.all_qubits();
203 for (unsigned i = 0; i < all_qubits.size(); i++) {
204 relabelling_map.insert({all_qubits[i], Qubit(label, i)});
205 }
206
207 circuit.rename_units(relabelling_map);
208 changed |= update_maps(maps, relabelling_map, relabelling_map);
209 return changed;
210 });
211 PredicatePtrMap precons = {};
212 PostConditions postcons = {};
213
214 nlohmann::json j;
215 j["name"] = "FlattenRelabelRegistersPass";
216 j["label"] = label;
217 return std::make_shared<StandardPass>(precons, t, postcons, j);
218}

240PassPtr gen_placement_pass(const Placement::Ptr& placement_ptr) {
241 Transform::Transformation trans = [=](Circuit& circ,
242 std::shared_ptr<unit_bimaps_t> maps) {
243 // Fall back to line placement if graph placement fails
244 bool changed;
245 try {
246 changed = placement_ptr->place(circ, maps);
247 } catch (const std::runtime_error& e) {
248 std::stringstream ss;
249 ss << "PlacementPass failed with message: " << e.what()
250 << " Fall back to LinePlacement.";
251 tket_log()->warn(ss.str());
252 Placement::Ptr line_placement_ptr = std::make_shared<LinePlacement>(
253 placement_ptr->get_architecture_ref());
254 changed = line_placement_ptr->place(circ, maps);
255 }
256 return changed;
257 };
258 Transform t = Transform(trans);
259 PredicatePtr twoqbpred = std::make_shared<MaxTwoQubitGatesPredicate>();
260 PredicatePtr n_qubit_pred = std::make_shared<MaxNQubitsPredicate>(
261 placement_ptr->get_architecture_ref().n_nodes());
262
263 PredicatePtrMap precons{
264 CompilationUnit::make_type_pair(twoqbpred),
265 CompilationUnit::make_type_pair(n_qubit_pred)};
266 PredicatePtr placement_pred = std::make_shared<PlacementPredicate>(
267 placement_ptr->get_architecture_ref());
268 PredicatePtrMap s_postcons{CompilationUnit::make_type_pair(placement_pred)};
269 PostConditions pc{s_postcons, {}, Guarantee::Preserve};
270 // record pass config
271 nlohmann::json j;
272 j["name"] = "PlacementPass";
273 j["placement"] = placement_ptr;
274 return std::make_shared<StandardPass>(precons, t, pc, j);
275}

335 const Architecture& arc, const std::vector<RoutingMethodPtr>& config) {
336 Transform::Transformation trans = [=](Circuit& circ,
337 std::shared_ptr<unit_bimaps_t> maps) {
338 MappingManager mm(std::make_shared<Architecture>(arc));
339 return mm.route_circuit_with_maps(circ, config, maps);
340 };
341 Transform t = Transform(trans);
342
343 PredicatePtr twoqbpred = std::make_shared<MaxTwoQubitGatesPredicate>();
344 PredicatePtr n_qubit_pred =
345 std::make_shared<MaxNQubitsPredicate>(arc.n_nodes());
346 PredicatePtrMap precons{
347 CompilationUnit::make_type_pair(twoqbpred),
348 CompilationUnit::make_type_pair(n_qubit_pred)};
349
350 PredicatePtr postcon1 = std::make_shared<ConnectivityPredicate>(arc);
351 std::pair<const std::type_index, PredicatePtr> pair1 =
352 CompilationUnit::make_type_pair(postcon1);
353 PredicatePtr postcon2 = std::make_shared<NoWireSwapsPredicate>();
354 PredicatePtrMap s_postcons{pair1, CompilationUnit::make_type_pair(postcon2)};
355 std::type_index gateset_ti = typeid(GateSetPredicate);
356 // clears all Routing predicates apart from the ones which are specified
357 // clears all GateSet predicates (inserts SWAPs and BRIDGEs)
358 PredicateClassGuarantees g_postcons{
359 {pair1.first, Guarantee::Clear},
360 {gateset_ti, Guarantee::Clear},
361 {typeid(MaxTwoQubitGatesPredicate), Guarantee::Clear}};
362 PostConditions pc{s_postcons, g_postcons, Guarantee::Preserve};
363
364 // record pass config
365 nlohmann::json j;
366 j["name"] = "RoutingPass";
367 j["routing_config"] = config;
368 j["architecture"] = arc;
369
370 return std::make_shared<StandardPass>(precons, t, pc, j);
371}

373PassPtr gen_placement_pass_phase_poly(
374 const Architecture& arc, unsigned maximum_matches, unsigned timeout,
375 unsigned maximum_pattern_gates, unsigned maximum_pattern_depth) {
376 Transform::Transformation trans = [=](Circuit& circ,
377 std::shared_ptr<unit_bimaps_t> maps) {
378 if (arc.n_nodes() < circ.n_qubits()) {
379 throw CircuitInvalidity(
380 "Circuit has more qubits than the architecture has nodes.");
381 }
382
383 qubit_vector_t q_vec = circ.all_qubits();
384 std::map<Qubit, Node> qubit_to_nodes;
385 unsigned counter = 0;
386
387 for (Node x : arc.nodes()) {
388 if (counter < circ.n_qubits()) {
389 qubit_to_nodes.insert({q_vec[counter], x});
390 ++counter;
391 }
392 }
393
394 circ.rename_units(qubit_to_nodes);
395 update_maps(maps, qubit_to_nodes, qubit_to_nodes);
396
397 return true;
398 };
399 Transform t = Transform(trans);
400
401 PredicatePtr no_wire_swap = std::make_shared<NoWireSwapsPredicate>();
402 PredicatePtrMap precons{CompilationUnit::make_type_pair(no_wire_swap)};
403
404 PredicatePtr placement_pred = std::make_shared<PlacementPredicate>(arc);
405 PredicatePtr n_qubit_pred =
406 std::make_shared<MaxNQubitsPredicate>(arc.n_nodes());
407
408 PredicatePtrMap s_postcons{
409 CompilationUnit::make_type_pair(placement_pred),
410 CompilationUnit::make_type_pair(n_qubit_pred),
411 CompilationUnit::make_type_pair(no_wire_swap)};
412
413 PostConditions pc{s_postcons, {}, Guarantee::Preserve};
414 // record pass config
415 nlohmann::json j;
416 j["name"] = "PlacementPass";
417 Placement::Ptr pp = std::make_shared<GraphPlacement>(
418 arc, maximum_matches, timeout, maximum_pattern_gates,
419 maximum_pattern_depth);
420 j["params"]["placement"] = pp;
421 return std::make_shared<StandardPass>(precons, t, pc, j);
422}

424PassPtr aas_routing_pass(
425 const Architecture& arc, const unsigned lookahead,
426 const aas::CNotSynthType cnotsynthtype) {
427 Transform::SimpleTransformation trans = [=](Circuit& circ) {
428 // check input:
429 if (lookahead == 0) {
430 throw std::logic_error("lookahead must be > 0");
431 }
432 if (arc.n_nodes() < circ.n_qubits()) {
433 throw CircuitInvalidity(
434 "Circuit has more qubits than the architecture has nodes.");
435 }
436
437 // this pass is not able to handle implicit wire swaps
438 // this is additionally assured by a precondition of this pass
439 TKET_ASSERT(!circ.has_implicit_wireswaps());
440
441 qubit_vector_t all_qu = circ.all_qubits();
442
443 Circuit input_circ = circ;
444
445 VertexList bin;
446
447 BGL_FORALL_VERTICES(v, circ.dag, DAG) {
448 if (!circ.detect_boundary_Op(v)) {
449 bin.push_back(v);
450 }
451 }
452
453 circ.remove_vertices(
454 bin, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::Yes);
455
456 for (const Command& com : input_circ) {
457 OpType ot = com.get_op_ptr()->get_type();
458 unit_vector_t qbs = com.get_args();
459 switch (ot) {
460 case OpType::PhasePolyBox: {
461 Op_ptr op = com.get_op_ptr();
462 const auto& ppb = dynamic_cast<const PhasePolyBox&>(*op);
463
464 Circuit result =
465 aas::phase_poly_synthesis(arc, ppb, lookahead, cnotsynthtype);
466
467 for (const Command& res_com : result) {
468 OpType ot = res_com.get_op_ptr()->get_type();
469 unit_vector_t res_qbs = res_com.get_args();
470
471 switch (ot) {
472 case OpType::CX: {
473 circ.add_op<Node>(ot, {Node(res_qbs[0]), Node(res_qbs[1])});
474 break;
475 }
476 case OpType::Rz: {
477 auto angle = res_com.get_op_ptr()->get_params().at(0);
478 circ.add_op<Node>(ot, angle, {Node(res_qbs[0])});
479 break;
480 }
481 default: {
482 TKET_ASSERT(!"Invalid gate type in phase poly box");
483 }
484 }
485 }
486 break;
487 }
488 case OpType::H: {
489 circ.add_op<Node>(ot, {Node(qbs[0])});
490 break;
491 }
492 case OpType::Collapse: {
493 circ.add_op<Node>(ot, {Node(qbs[0])});
494 break;
495 }
496 case OpType::Reset: {
497 circ.add_op<Node>(ot, {Node(qbs[0])});
498 break;
499 }
500 case OpType::Measure: {
501 circ.add_measure(Node(qbs[0]), Bit(qbs[1]));
502 break;
503 }
504 default: {
505 throw BadOpType("Invalid gate in input of AAS routing", ot);
506 }
507 }
508 }
509
510 return true;
511 };
512 Transform t = Transform(trans);
513
514 PredicatePtr placedpred = std::make_shared<PlacementPredicate>(arc);
515 PredicatePtr n_qubit_pred =
516 std::make_shared<MaxNQubitsPredicate>(arc.n_nodes());
517 PredicatePtr no_wire_swap = std::make_shared<NoWireSwapsPredicate>();
518
519 PredicatePtrMap precons{
520 CompilationUnit::make_type_pair(placedpred),
521 CompilationUnit::make_type_pair(n_qubit_pred),
522 CompilationUnit::make_type_pair(no_wire_swap)};
523
524 PredicatePtr postcon1 = std::make_shared<ConnectivityPredicate>(arc);
525 std::pair<const std::type_index, PredicatePtr> pair1 =
526 CompilationUnit::make_type_pair(postcon1);
527 PredicatePtr postcon2 = std::make_shared<NoWireSwapsPredicate>();
528 PredicatePtrMap s_postcons{pair1, CompilationUnit::make_type_pair(postcon2)};
529 std::type_index gateset_ti = typeid(GateSetPredicate);
530 PredicateClassGuarantees g_postcons{
531 {pair1.first, Guarantee::Clear}, {gateset_ti, Guarantee::Clear}};
532 PostConditions pc{s_postcons, g_postcons, Guarantee::Preserve};
533
534 nlohmann::json j;
535 j["name"] = "AASRoutingPass";
536 j["architecture"] = arc;
537
538 return std::make_shared<StandardPass>(precons, t, pc, j);
539}

566PassPtr gen_decompose_routing_gates_to_cxs_pass(
567 const Architecture& arc, bool directed) {
568 PredicateClassGuarantees g_postcons{
569 {typeid(GateSetPredicate), Guarantee::Clear}};
570 PredicatePtrMap precons;
571 PredicatePtrMap s_postcons;
572 Transform t = Transforms::decompose_SWAP_to_CX(arc) >>
573 Transforms::decompose_BRIDGE_to_CX() >>
574 Transforms::remove_redundancies();
575 if (directed) {
576 OpTypeSet out_optypes{all_single_qubit_types()};
577 out_optypes.insert(OpType::CX);
578 OpTypeSet in_optypes = out_optypes;
579 in_optypes.insert(OpType::SWAP);
580 in_optypes.insert(OpType::BRIDGE);
581 PredicatePtr twoqbpred = std::make_shared<MaxTwoQubitGatesPredicate>();
582 PredicatePtr connected = std::make_shared<ConnectivityPredicate>(arc);
583 PredicatePtr wireswaps = std::make_shared<NoWireSwapsPredicate>();
