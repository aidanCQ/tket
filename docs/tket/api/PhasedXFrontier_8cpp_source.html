31template <typename T>
32static std::map<T, unsigned> count(std::vector<T> xs) {
33 std::map<T, unsigned> cnts;
34 for (const auto& x : xs) {
35 auto it = cnts.find(x);
36 if (it != cnts.end()) {
37 ++(it->second);
38 } else {
39 cnts[x] = 1;
40 }
41 }
42 return cnts;
43}

71OptEdge PhasedXFrontier::get_beta_edge(unsigned i) const {
72 const auto& [start, end] = intervals_[i];
73 Edge e = start;
74 while (e != end) {
75 Vertex v = circ_.target(e);
76 OpType type = circ_.get_OpType_from_Vertex(v);
77 if (type == OpType::PhasedX || type == OpType::NPhasedX) {
78 return e;
79 }
80 e = circ_.get_next_edge(v, e);
81 }
82 return std::nullopt;
83}

85OptEdgeVec PhasedXFrontier::get_all_beta_edges() const {
86 OptEdgeVec beta_edges;
87 std::vector<Vertex> beta_vertices;
88 for (unsigned i = 0; i < circ_.n_qubits(); ++i) {
89 OptEdge beta_e = get_beta_edge(i);
90 beta_edges.push_back(beta_e);
91 if (beta_e) {
92 beta_vertices.push_back(circ_.target(*beta_e));
93 }
94 }
95
96 // for OpType::NPhasedX, we need to reset some betas to zero if they
97 // are shadowed (i.e. if one NPhasedX is behind another one)
98 std::map<Vertex, unsigned> arities = count(beta_vertices);
99 for (auto& e : beta_edges) {
100 if (e) {
101 Vertex v = circ_.target(*e);
102 Op_ptr op = circ_.get_Op_ptr_from_Vertex(v);
103 if (arities[v] != op->n_qubits()) {
104 e = std::nullopt;
105 }
106 }
107 }
108 return beta_edges;
109}

125std::vector<Expr> PhasedXFrontier::get_all_betas() const {

154Edge PhasedXFrontier::get_interval_end(Edge e) const {
155 Vertex v = circ_.target(e);
156 while (!circ_.detect_final_Op(v) && !is_interval_boundary(v)) {
157 auto p = circ_.get_next_pair(v, e);
158 v = p.first;
159 e = p.second;
160 }
161 return e;
162}

172bool PhasedXFrontier::is_interval_boundary(Op_ptr op) {
173 OpType type = op->get_type();
174 if (type == OpType::Conditional) {
175 // We currently split intervals on Conditional gates even if they
176 // are single-qubit gates, as there is no simple way to guarantee that
177 // classically controlled gates are space-like separated without making
178 // additional assumptions or walking the DAG.
179 return true;
180 }
181 return is_gate_type(type) && as_gate_ptr(op)->n_qubits() > 1 &&
182 type != OpType::NPhasedX;
183}

207PhasedXFrontier::PhasedXFrontier(Circuit& circ)
208 : intervals_(),
209 circ_(circ),
210 squasher_(std::make_unique<PhasedXSquasher>(), circ, false) {
211 const unsigned n = circ_.n_qubits();
212 intervals_.resize(n);
213
214 // initialise intervals
215 const qubit_vector_t all_qbs = circ_.all_qubits();
216 for (unsigned i = 0; i < n; ++i) {
217 Qubit q = all_qbs[i];
218 Vertex v_in = circ_.get_in(q);
219 EdgeVec e_vec = circ_.get_all_out_edges(v_in);
220 TKET_ASSERT(e_vec.size() == 1);
221
222 // interval [start, end]
223 Edge start = e_vec.front();
224 Edge end = get_interval_end(start);
225 intervals_[i] = {start, end};
226 }
227}

229void PhasedXFrontier::squash_interval(unsigned i) {
230 auto& [start_e, end_e] = intervals_[i];
231
232 // backup edges
233 VertPort start{circ_.source(start_e), circ_.get_source_port(start_e)};
234 VertPort end{circ_.target(end_e), circ_.get_target_port(end_e)};
235
236 squasher_.squash_between(start_e, end_e);
237
238 // restore interval edges
239 start_e = circ_.get_nth_out_edge(start.first, start.second);
240 end_e = circ_.get_nth_in_edge(end.first, end.second);
241}

243PhasedXFrontier::BackupIntervals PhasedXFrontier::backup_intervals() const {
244 BackupIntervals ret;
245 for (unsigned i = 0; i < circ_.n_qubits(); ++i) {
246 const auto& [start, end] = intervals_[i];
247 ret.start.push_back({circ_.source(start), circ_.get_source_port(start)});
248 ret.end.push_back({circ_.target(end), circ_.get_target_port(end)});
249 }
250 return ret;
251}

253void PhasedXFrontier::restore_intervals(const BackupIntervals& b) {
254 for (unsigned i = 0; i < circ_.n_qubits(); ++i) {
255 Edge start = circ_.get_nth_out_edge(b.start[i].first, b.start[i].second);
256 Edge end = circ_.get_nth_in_edge(b.end[i].first, b.end[i].second);
257 intervals_[i].first = start;
258 intervals_[i].second = end;
259 }

262void PhasedXFrontier::skip_global_gates(unsigned n) {
263 for (unsigned i = 0; i < circ_.n_qubits(); ++i) {
264 unsigned count = 0;
265 auto& [e, end] = intervals_[i];
266 while (e != end) {
267 Vertex v = circ_.target(e);
268 OpType type = circ_.get_OpType_from_Vertex(v);
269 e = circ_.get_next_edge(v, e);
270 if (type == OpType::NPhasedX ||
271 (circ_.n_qubits() == 1 && type == OpType::PhasedX)) {
272 unsigned in_edges = circ_.n_in_edges_of_type(v, EdgeType::Quantum);
273 unsigned out_edges = circ_.n_out_edges_of_type(v, EdgeType::Quantum);
274 if (in_edges != circ_.n_qubits() || out_edges != circ_.n_qubits()) {
275 throw CircuitInvalidity("Found non-global NPhasedX gate");
276 }
277 ++count;
278 if (count == n) {
279 break;
280 }
281 }
282 }
283 if (count < n) {
284 throw CircuitInvalidity("Did not find expected global gates");
285 }
286 }

289bool PhasedXFrontier::are_phasedx_left() const {
290 PhasedXFrontier copy = *this;
291 const unsigned n = circ_.n_qubits();
292 for (unsigned i = 0; i < n; ++i) {
293 copy.next_interval(i);

295 OptVertexVec all_phasedx = copy.get_all_beta_vertices();

300void PhasedXFrontier::insert_1_phasedx(unsigned i) {
301 std::vector<Expr> betas = get_all_betas();
302 OptEdgeVec edges = get_all_beta_edges();
303 OptVertexVec vertices = get_all_beta_vertices();
304
305 if (!vertices[i]) {
306 throw CircuitInvalidity("No PhasedX found on qubit " + std::to_string(i));
307 }
308 Expr beta = betas[i];
309
310 VertexSet bin;
311 EdgeVec in_hole, out_hole;
312 Circuit sub1(circ_.n_qubits());
313 Circuit sub2(circ_.n_qubits());
314
315 for (unsigned i = 0; i < circ_.n_qubits(); ++i) {
316 if (vertices[i]) {
317 Vertex v = vertices[i].value();
318 Edge e = edges[i].value();
319 Op_ptr op = circ_.get_Op_ptr_from_Vertex(v);
320 OpType type = op->get_type();
321 Expr new_beta = betas[i] - beta;
322
323 in_hole.push_back(e);
324 out_hole.push_back(circ_.get_next_edge(v, e));
325
326 if (!bin.contains(v)) {
327 bin.insert(v);
328
329 // v's qubits
330 std::vector<unsigned> qubits;
331 for (unsigned j = 0; j < circ_.n_qubits(); ++j) {
332 if (vertices[j] == v) {
333 qubits.push_back(j);
334 }
335 }
336
337 if (type == OpType::NPhasedX || type == OpType::PhasedX) {
338 Expr alpha = op->get_params()[1];
339 if (!equiv_0(new_beta, 4)) {
340 if (equiv_0(new_beta, 2)) {
341 if (qubits.size() % 2) {
342 sub1.add_phase(-1);
343 }
344 } else {
345 sub2.add_op(type, {new_beta, 0.}, qubits);
346 }
347 }
348 if (!equiv_0(alpha, 2)) {
349 for (unsigned q : qubits) {
350 sub1.add_op<unsigned>(OpType::Rz, {-alpha}, {q});
351 sub2.add_op<unsigned>(OpType::Rz, {alpha}, {q});
352 }
353 }
354 } else {
355 throw BadOpType("Encountered invalid beta angle OpType", type);
356 }
357 }
358 } else {
359 Edge interval_begin = intervals_[i].first;
360 in_hole.push_back(interval_begin);
361 out_hole.push_back(interval_begin);
362 if (!equiv_0(beta, 2)) {
363 sub2.add_op<unsigned>(OpType::PhasedX, {-beta, 0}, {i});
364 } else if (!equiv_0(beta, 4)) {
365 sub2.add_phase(-1);
366 }
367 }
368 }
369
370 Circuit sub(circ_.n_qubits());
371 sub.append(sub1);
372 sub.add_op<Qubit>(OpType::NPhasedX, {beta, 0}, sub.all_qubits());
373 sub.append(sub2);
374
375 Subcircuit hole{in_hole, out_hole, bin};
376
377 // perform substitution
378 BackupIntervals backup = backup_intervals();
379 circ_.substitute(sub, hole, Circuit::VertexDeletion::Yes);
380 restore_intervals(backup);
381
382 skip_global_gates(1);
383}

385void PhasedXFrontier::insert_2_phasedx() {
386 EdgeVec hole_in, hole_out;
387 Circuit sub1(circ_.n_qubits());
388 Circuit sub2(circ_.n_qubits());
389 Circuit sub3(circ_.n_qubits());
390 VertexSet bin;
391
392 std::vector<Expr> betas = get_all_betas();
393 OptEdgeVec edges = get_all_beta_edges();
394 OptVertexVec vertices = get_all_beta_vertices();
395
396 for (unsigned i = 0; i < circ_.n_qubits(); ++i) {
397 if (vertices[i]) {
398 Vertex v = vertices[i].value();
399 Edge e = edges[i].value();
400 Op_ptr op = circ_.get_Op_ptr_from_Vertex(v);
401 OpType type = op->get_type();
402 hole_in.push_back(e);
403 hole_out.push_back(circ_.get_next_edge(v, e));
404 bin.insert(v);
405 Expr beta = betas[i];
406 if (type == OpType::NPhasedX || type == OpType::PhasedX) {
407 Expr alpha = op->get_params()[1];
408 if (!equiv_0(alpha, 2)) {
409 sub1.add_op<unsigned>(OpType::Rz, {-alpha}, {i});
410 sub3.add_op<unsigned>(OpType::Rz, {alpha}, {i});
411 }
412 }
413 if (!equiv_0(beta, 2)) {
414 sub2.add_op<unsigned>(OpType::Rz, {beta}, {i});
415 } else if (!equiv_0(beta, 4)) {
416 sub2.add_phase(-1);
417 }
418 } else {
419 Edge interval_begin = intervals_[i].first;
420 hole_in.push_back(interval_begin);
421 hole_out.push_back(interval_begin);
422 }
423 }
424
425 Circuit sub(circ_.n_qubits());
426 sub.append(sub1);
427 sub.add_op(OpType::NPhasedX, {-0.5, 0.5}, sub.all_qubits());
428 sub.append(sub2);
429 sub.add_op(OpType::NPhasedX, {0.5, 0.5}, sub.all_qubits());
430 sub.append(sub3);
431 Subcircuit hole{hole_in, hole_out, bin};
432
433 BackupIntervals backup = backup_intervals();
434 circ_.substitute(sub, hole, Circuit::VertexDeletion::Yes);
435 restore_intervals(backup);
436
437 skip_global_gates(2);
438}
