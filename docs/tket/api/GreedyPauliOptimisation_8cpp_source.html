35static void apply_tqe_to_circ(const TQE& tqe, Circuit& circ) {
36 auto [gate_type, a, b] = tqe;
37 switch (gate_type) {
38 case TQEType::XX:
39 circ.add_op<unsigned>(OpType::H, {a});
40 circ.add_op<unsigned>(OpType::CX, {a, b});
41 circ.add_op<unsigned>(OpType::H, {a});
42 break;
43 case TQEType::XY:
44 circ.add_op<unsigned>(OpType::H, {a});
45 circ.add_op<unsigned>(OpType::CY, {a, b});
46 circ.add_op<unsigned>(OpType::H, {a});
47 break;
48 case TQEType::XZ:
49 circ.add_op<unsigned>(OpType::CX, {b, a});
50 break;
51 case TQEType::YX:
52 circ.add_op<unsigned>(OpType::H, {b});
53 circ.add_op<unsigned>(OpType::CY, {b, a});
54 circ.add_op<unsigned>(OpType::H, {b});
55 break;
56 case TQEType::YY:
57 circ.add_op<unsigned>(OpType::V, {a});
58 circ.add_op<unsigned>(OpType::CY, {a, b});
59 circ.add_op<unsigned>(OpType::Vdg, {a});
60 break;
61 case TQEType::YZ:
62 circ.add_op<unsigned>(OpType::CY, {b, a});
63 break;
64 case TQEType::ZX:
65 circ.add_op<unsigned>(OpType::CX, {a, b});
66 break;
67 case TQEType::ZY:
68 circ.add_op<unsigned>(OpType::CY, {a, b});
69 break;
70 case TQEType::ZZ:
71 circ.add_op<unsigned>(OpType::CZ, {a, b});
72 break;
73 }
74}

76static void apply_tqe_to_tableau(const TQE& tqe, UnitaryRevTableau& tab) {
77 auto [gate_type, a_int, b_int] = tqe;
78 Qubit a(a_int);
79 Qubit b(b_int);
80 switch (gate_type) {
81 case TQEType::XX:
82 tab.apply_gate_at_end(OpType::H, {a});
83 tab.apply_gate_at_end(OpType::CX, {a, b});
84 tab.apply_gate_at_end(OpType::H, {a});
85 break;
86 case TQEType::XY:
87 tab.apply_gate_at_end(OpType::H, {a});
88 tab.apply_gate_at_end(OpType::CY, {a, b});
89 tab.apply_gate_at_end(OpType::H, {a});
90 break;
91 case TQEType::XZ:
92 tab.apply_gate_at_end(OpType::CX, {b, a});
93 break;
94 case TQEType::YX:
95 tab.apply_gate_at_end(OpType::H, {b});
96 tab.apply_gate_at_end(OpType::CY, {b, a});
97 tab.apply_gate_at_end(OpType::H, {b});
98 break;
99 case TQEType::YY:
100 tab.apply_gate_at_end(OpType::V, {a});
101 tab.apply_gate_at_end(OpType::CY, {a, b});
102 tab.apply_gate_at_end(OpType::Vdg, {a});
103 break;
104 case TQEType::YZ:
105 tab.apply_gate_at_end(OpType::CY, {b, a});
106 break;
107 case TQEType::ZX:
108 tab.apply_gate_at_end(OpType::CX, {a, b});
109 break;
110 case TQEType::ZY:
111 tab.apply_gate_at_end(OpType::CY, {a, b});
112 break;
113 case TQEType::ZZ:
114 tab.apply_gate_at_end(OpType::CZ, {a, b});
115 break;
116 }

125int PauliExpNode::tqe_cost_increase(const TQE& tqe) const {
126 unsigned supp0 = support_vec_[std::get<1>(tqe)];
127 unsigned supp1 = support_vec_[std::get<2>(tqe)];
128 unsigned new_supp0, new_supp1;
129 std::tie(new_supp0, new_supp1) =
130 SINGLET_PAIR_TRANSFORMATION_MAP.at({std::get<0>(tqe), supp0, supp1});
131 return (new_supp0 > 0) + (new_supp1 > 0) - (supp0 > 0) - (supp1 > 0);
132}

134void PauliExpNode::update(const TQE& tqe) {
135 unsigned a = std::get<1>(tqe);
136 unsigned b = std::get<2>(tqe);
137 unsigned supp0 = support_vec_[a];
138 unsigned supp1 = support_vec_[b];
139 unsigned new_supp0, new_supp1;
140 std::tie(new_supp0, new_supp1) =
141 SINGLET_PAIR_TRANSFORMATION_MAP.at({std::get<0>(tqe), supp0, supp1});
142 support_vec_[a] = new_supp0;
143 support_vec_[b] = new_supp1;
144 tqe_cost_ += (new_supp0 > 0) + (new_supp1 > 0) - (supp0 > 0) - (supp1 > 0);

147std::vector<TQE> PauliExpNode::reduction_tqes() const {
148 std::vector<TQE> tqes;
149 // qubits with support
150 std::vector<unsigned> sqs;
151 for (unsigned i = 0; i < support_vec_.size(); i++) {
152 if (support_vec_[i] > 0) sqs.push_back(i);
153 }
154 for (unsigned i = 0; i < sqs.size() - 1; i++) {
155 for (unsigned j = i + 1; j < sqs.size(); j++) {
156 std::vector<TQEType> tqe_types = SINGLET_PAIR_REDUCTION_TQES.at(
157 {support_vec_[sqs[i]], support_vec_[sqs[j]]});
158 for (const TQEType& tt : tqe_types) {
159 tqes.push_back({tt, sqs[i], sqs[j]});
160 }
161 }
162 }
163 return tqes;
164}

176TableauRowNode::TableauRowNode(std::vector<unsigned> support_vec)
177 : support_vec_(support_vec) {
178 n_weaks_ = 0;
179 n_strongs_ = 0;
180 for (const unsigned& supp : support_vec_) {
181 SupportType st = FACTOR_WEAKNESS_MAP.at(supp);
182 if (st == SupportType::Strong) {
183 n_strongs_++;
184 } else if (st == SupportType::Weak) {
185 n_weaks_++;
186 }
187 }
188 tqe_cost_ = static_cast<unsigned>(1.5 * (n_strongs_ - 1) + n_weaks_);
189}

191int TableauRowNode::tqe_cost_increase(const TQE& tqe) const {
192 unsigned supp0 = support_vec_[std::get<1>(tqe)];
193 unsigned supp1 = support_vec_[std::get<2>(tqe)];
194 unsigned new_supp0, new_supp1;
195 std::tie(new_supp0, new_supp1) =
196 FACTOR_PAIR_TRANSFORMATION_MAP.at({std::get<0>(tqe), supp0, supp1});
197 SupportType st_supp0 = FACTOR_WEAKNESS_MAP.at(supp0);
198 SupportType st_supp1 = FACTOR_WEAKNESS_MAP.at(supp1);
199 SupportType st_new_supp0 = FACTOR_WEAKNESS_MAP.at(new_supp0);
200 SupportType st_new_supp1 = FACTOR_WEAKNESS_MAP.at(new_supp1);
201 unsigned old_strongs =
202 (st_supp0 == SupportType::Strong) + (st_supp1 == SupportType::Strong);
203 unsigned old_weaks =
204 (st_supp0 == SupportType::Weak) + (st_supp1 == SupportType::Weak);
205 unsigned new_strongs = (st_new_supp0 == SupportType::Strong) +
206 (st_new_supp1 == SupportType::Strong);
207 unsigned new_weaks =
208 (st_new_supp0 == SupportType::Weak) + (st_new_supp1 == SupportType::Weak);
209 int strong_increase = new_strongs - old_strongs;
210 int weak_increase = new_weaks - old_weaks;
211 return static_cast<int>(1.5 * strong_increase + weak_increase);
212}

241std::vector<TQE> TableauRowNode::reduction_tqes() const {
242 std::vector<TQE> tqes;
243 // qubits with support
244 std::vector<unsigned> sqs;
245 for (unsigned i = 0; i < support_vec_.size(); i++) {
246 if (support_vec_[i] > 0) sqs.push_back(i);
247 }
248 for (unsigned i = 0; i < sqs.size() - 1; i++) {
249 for (unsigned j = i + 1; j < sqs.size(); j++) {
250 std::vector<TQEType> tqe_types;
251 unsigned a = sqs[i];
252 unsigned b = sqs[j];
253 unsigned supp0 = support_vec_[a];
254 unsigned supp1 = support_vec_[b];
255 SupportType st_supp0 = FACTOR_WEAKNESS_MAP.at(supp0);
256 SupportType st_supp1 = FACTOR_WEAKNESS_MAP.at(supp1);
257 if (st_supp0 == SupportType::Strong) {
258 if (st_supp1 == SupportType::Strong) {
259 // TQEs that transform a SS pair to WW
260 tqe_types = FACTOR_PAIR_SS_TO_WW_TQES.at({supp0, supp1});
261 } else {
262 // TQEs that transform a SW pair to a single strong
263 tqe_types = FACTOR_PAIR_SW_TO_SN_TQES.at({supp0, supp1});
264 }
265 } else {
266 if (st_supp1 == SupportType::Strong) {
267 // TQEs that transform a WS pair to a single strong
268 tqe_types = FACTOR_PAIR_SW_TO_SN_TQES.at({supp1, supp0});
269 // flip qubits
270 a = sqs[j];
271 b = sqs[i];
272 } else {
273 // TQEs that transform a WW pair to a single weak, not always
274 // possible
275 tqe_types = FACTOR_PAIR_WW_TO_WN_OR_NW_TQES.at({supp0, supp1});
276 }
277 }
278 for (const TQEType& tt : tqe_types) {
279 tqes.push_back({tt, a, b});
280 }
281 }
282 }
283 return tqes;
284}

296// return the sum of the cost increases on remaining tableau nodes
297static double default_tableau_tqe_cost(
298 const std::vector<TableauRowNode>& rows,
299 const std::vector<unsigned>& remaining_indices, const TQE& tqe) {
300 double cost = 0;
301 for (const unsigned& index : remaining_indices) {
302 cost += rows[index].tqe_cost_increase(tqe);
303 }
304 return cost;
305}

307// return the weighted sum of the cost increases on remaining nodes
308// we discount the weight after each set
309static double default_pauliexp_tqe_cost(
310 const double discount_rate,
311 const std::vector<std::vector<PauliExpNode>>& rotation_sets,
312 const std::vector<TableauRowNode>& rows, const TQE& tqe) {
313 double discount = 1 / (1 + discount_rate);
314 double weight = 1;
315 double exp_cost = 0;
316 double tab_cost = 0;
317 for (const std::vector<PauliExpNode>& rotation_set : rotation_sets) {
318 for (const PauliExpNode& node : rotation_set) {
319 exp_cost += weight * node.tqe_cost_increase(tqe);
320 }
321 weight *= discount;
322 }
323 for (const TableauRowNode& node : rows) {
324 tab_cost += weight * node.tqe_cost_increase(tqe);
325 }
326 return exp_cost + tab_cost;
327}

329// given a map from TQE to a vector of costs, select the one with the minimum
330// weighted sum of minmax-normalised costs
331static TQE minmax_selection(
332 const std::map<TQE, std::vector<double>>& tqe_candidates_cost,
333 const std::vector<double>& weights) {
334 TKET_ASSERT(tqe_candidates_cost.size() > 0);
335 size_t n_costs = tqe_candidates_cost.begin()->second.size();
336 TKET_ASSERT(n_costs == weights.size());
337 // for each cost type, store its min and max
338 std::vector<double> mins = tqe_candidates_cost.begin()->second;
339 std::vector<double> maxs = tqe_candidates_cost.begin()->second;
340 for (const auto& pair : tqe_candidates_cost) {
341 TKET_ASSERT(pair.second.size() == n_costs);
342 for (unsigned cost_index = 0; cost_index < n_costs; cost_index++) {
343 if (pair.second[cost_index] < mins[cost_index]) {
344 mins[cost_index] = pair.second[cost_index];
345 }
346 if (pair.second[cost_index] > maxs[cost_index]) {
347 maxs[cost_index] = pair.second[cost_index];
348 }
349 }
350 }
351 // valid_indices stores the indices of the costs where min!=max
352 std::vector<unsigned> valid_indices;
353 for (unsigned cost_index = 0; cost_index < n_costs; cost_index++) {
354 if (mins[cost_index] != maxs[cost_index]) {
355 valid_indices.push_back(cost_index);
356 }
357 }
358 // if all have the same cost, return the first one
359 if (valid_indices.size() == 0) {
360 TQE min_tqe = tqe_candidates_cost.begin()->first;
361 return min_tqe;
362 }
363 // if only one cost variable, no need to normalise
364 if (valid_indices.size() == 1) {
365 auto it = tqe_candidates_cost.begin();
366 double min_cost = it->second[valid_indices[0]];
367 TQE min_tqe = it->first;
368 for (; it != tqe_candidates_cost.end(); it++) {
369 if (it->second[valid_indices[0]] < min_cost) {
370 min_tqe = it->first;
371 min_cost = it->second[valid_indices[0]];
372 }
373 }
374 return min_tqe;
375 }
376 // find the tqe with the minimum normalised cost
377 auto it = tqe_candidates_cost.begin();
378 double min_cost = 0;
379 TQE min_tqe = it->first;
380 // initialise min_cost
381 for (const auto& cost_index : valid_indices) {
382 min_cost += weights[cost_index] *
383 (it->second[cost_index] - mins[cost_index]) /
384 (maxs[cost_index] - mins[cost_index]);
385 }
386 it++;
387 // iterate all tqes
388 for (; it != tqe_candidates_cost.end(); it++) {
389 double cost = 0;
390 for (const auto& cost_index : valid_indices) {
391 cost += weights[cost_index] *
392 (it->second[cost_index] - mins[cost_index]) /
393 (maxs[cost_index] - mins[cost_index]);
394 }
395 if (cost < min_cost) {
396 min_cost = cost;
397 min_tqe = it->first;
398 }
399 }
400 return min_tqe;
401}

403static TQE select_pauliexp_tqe(
404 const std::map<TQE, std::vector<double>>& tqe_candidates_cost,
405 double depth_weight) {
406 return minmax_selection(tqe_candidates_cost, {1, depth_weight});

410 const std::map<TQE, std::vector<double>>& tqe_candidates_cost,

415// simple struct that tracks the depth on each qubit
416struct DepthTracker {
417 std::vector<unsigned> qubit_depth;
418 unsigned max_depth;
419 DepthTracker(unsigned n) : qubit_depth(n, 0), max_depth(0) {};
420
421 unsigned gate_depth(unsigned a, unsigned b) const {
422 return std::max(qubit_depth[a], qubit_depth[b]) + 1;
423 };
424 void add_1q_gate(unsigned a) {
425 qubit_depth[a]++;
426 if (qubit_depth[a] > max_depth) {
427 max_depth = qubit_depth[a];
428 }
429 };
430 void add_2q_gate(unsigned a, unsigned b) {
431 unsigned new_gate_depth = gate_depth(a, b);
432 qubit_depth[a] = new_gate_depth;
433 qubit_depth[b] = new_gate_depth;
434 if (new_gate_depth > max_depth) {
435 max_depth = new_gate_depth;
436 }
437 };
438};

445static void tableau_cleanup(
446 std::vector<TableauRowNode>& rows, UnitaryRevTableau& tab, Circuit& circ) {
447 // apply local Cliffords
448 for (const TableauRowNode& node : rows) {
449 unsigned q_index, supp;
450 std::tie(q_index, supp) = node.first_support();
451 Qubit q(q_index);
452 std::vector<LocalCliffordType> local_cliffords =
453 FACTOR_STRONG_TO_LOCALS.at(supp);
454 for (const LocalCliffordType& lc : local_cliffords) {
455 switch (lc) {
456 case LocalCliffordType::H:
457 tab.apply_gate_at_end(OpType::H, {q});
458 circ.add_op<UnitID>(OpType::H, {q});
459 break;
460 case LocalCliffordType::S:
461 tab.apply_gate_at_end(OpType::S, {q});
462 circ.add_op<UnitID>(OpType::S, {q});
463 break;
464 case LocalCliffordType::V:
465 tab.apply_gate_at_end(OpType::V, {q});
466 circ.add_op<UnitID>(OpType::V, {q});
467 break;
468 }
469 }
470 }
471 // remove signs
472 for (const Qubit& q : circ.all_qubits()) {
473 if (cast_coeff<quarter_turns_t, Complex>(tab.get_xrow(q).coeff) != 1.) {
474 tab.apply_gate_at_end(OpType::Z, {q});
475 circ.add_op<UnitID>(OpType::Z, {q});
476 }
477 if (cast_coeff<quarter_turns_t, Complex>(tab.get_zrow(q).coeff) != 1.) {
478 tab.apply_gate_at_end(OpType::X, {q});
479 circ.add_op<UnitID>(OpType::X, {q});
480 }
481 }
482 // remove permutations
483 // 1. find perm
484 unsigned n_qubits = circ.n_qubits();
485 std::vector<unsigned> perm(n_qubits);
486 for (unsigned i = 0; i < n_qubits; i++) {
487 QubitPauliMap z_row_string = tab.get_zrow(Qubit(i)).string;
488 for (auto it = z_row_string.begin(); it != z_row_string.end(); it++) {
489 if (it->second == Pauli::Z) {
490 perm[it->first.index()[0]] = i;
491 break;
492 }
493 }
494 }
495 // 2. traverse transpositions
496 std::unordered_set<unsigned> done;
497 for (unsigned k = 0; k < n_qubits; k++) {
498 if (done.find(k) != done.end()) {
499 continue;
500 }
501 unsigned head = k;
502 unsigned current = k;
503 unsigned next = perm[k];
504 while (true) {
505 if (next == head) {
506 done.insert(current);
507 break;
508 }
509 // the SWAP gates will be later converted to wire swaps
510 tab.apply_gate_at_end(OpType::SWAP, {Qubit(current), Qubit(next)});
511 circ.add_op<unsigned>(OpType::SWAP, {current, next});
512 done.insert(current);
513 current = next;
514 next = perm[current];
515 }
516 }
517}

522static void tableau_row_nodes_synthesis(
523 std::vector<TableauRowNode>& rows, UnitaryRevTableau& tab, Circuit& circ,
524 double depth_weight, DepthTracker& depth_tracker) {
525 // only consider nodes with a non-zero cost
526 std::vector<unsigned> remaining_indices;
527 for (unsigned i = 0; i < rows.size(); i++) {
528 if (rows[i].tqe_cost() > 0) {
529 remaining_indices.push_back(i);
530 }
531 }
532 while (remaining_indices.size() != 0) {
533 // get nodes with min cost
534 std::vector<unsigned> min_nodes_indices = {remaining_indices[0]};
535 unsigned min_cost = rows[remaining_indices[0]].tqe_cost();
536 for (unsigned i = 1; i < remaining_indices.size(); i++) {
537 unsigned node_cost = rows[remaining_indices[i]].tqe_cost();
538 if (node_cost == min_cost) {
539 min_nodes_indices.push_back(remaining_indices[i]);
540 } else if (node_cost < min_cost) {
541 min_nodes_indices = {remaining_indices[i]};
542 min_cost = node_cost;
543 }
544 }
545 // for each node with min cost, find the list of tqe gates that can reduce
546 // its cost
547 std::set<TQE> tqe_candidates;
548 TKET_ASSERT(min_nodes_indices.size() > 0);
549 for (const unsigned& index : min_nodes_indices) {
550 std::vector<TQE> node_reducing_tqes = rows[index].reduction_tqes();
551 tqe_candidates.insert(
552 node_reducing_tqes.begin(), node_reducing_tqes.end());
553 }
554 // for each tqe we compute a vector of cost factors which will
555 // be combined to make the final decision.
556 // we currently only consider tqe_cost and gate_depth.
557 std::map<TQE, std::vector<double>> tqe_candidates_cost;
558 for (const TQE& tqe : tqe_candidates) {
559 tqe_candidates_cost.insert(
560 {tqe,
561 {default_tableau_tqe_cost(rows, remaining_indices, tqe),
562 static_cast<double>(depth_tracker.gate_depth(
563 std::get<1>(tqe), std::get<2>(tqe)))}});
564 }
565 TKET_ASSERT(tqe_candidates_cost.size() > 0);
566 // select the best one
567 TQE selected_tqe = select_tableau_tqe(tqe_candidates_cost, depth_weight);
568 // apply TQE
569 apply_tqe_to_circ(selected_tqe, circ);
570 apply_tqe_to_tableau(selected_tqe, tab);
571 // update depth tracker
572 depth_tracker.add_2q_gate(
573 std::get<1>(selected_tqe), std::get<2>(selected_tqe));
574 // remove finished nodes
575 for (unsigned i = remaining_indices.size(); i-- > 0;) {
576 unsigned node_index = remaining_indices[i];
577 rows[node_index].update(selected_tqe);
578 if (rows[node_index].tqe_cost() == 0) {
579 remaining_indices.erase(remaining_indices.begin() + i);
580 }
581 }
582 }
583 tableau_cleanup(rows, tab, circ);
584}

597static bool consume_available_rotations(
598 std::vector<std::vector<PauliExpNode>>& rotation_sets,
599 UnitaryRevTableau& tab, Circuit& circ, DepthTracker& depth_tracker) {
600 std::vector<unsigned> bin;
601 if (rotation_sets.size() == 0) {
602 return false;
603 }
604 std::vector<PauliExpNode>& first_set = rotation_sets[0];
605 for (unsigned i = 0; i < first_set.size(); i++) {
606 PauliExpNode& node = first_set[i];
607 if (node.tqe_cost() > 0) continue;
608 unsigned q_index, supp;
609 std::tie(q_index, supp) = node.first_support();
610 Qubit q(q_index);
611 depth_tracker.add_1q_gate(q_index);
612 switch (supp) {
613 case 3: {
614 // we apply S gate only to the frame, then check the sign, then Sdg
615 // if + apply f.Sdg; circ.Ry(-a)
616 // if - apply f.Sdg; circ.Ry(a)
617 tab.apply_gate_at_end(OpType::S, {q});
618 Complex x_coeff =
619 cast_coeff<quarter_turns_t, Complex>(tab.get_xrow(q).coeff);
620 tab.apply_gate_at_end(OpType::Sdg, {q});
621 if (x_coeff == 1.) {
622 circ.add_op<UnitID>(OpType::Ry, -node.theta(), {q});
623 } else {
624 circ.add_op<UnitID>(OpType::Ry, node.theta(), {q});
625 }
626 break;
627 }
628 case 1: {
629 Complex z_coeff =
630 cast_coeff<quarter_turns_t, Complex>(tab.get_zrow(q).coeff);
631 if (z_coeff == 1.) {
632 circ.add_op<UnitID>(OpType::Rz, node.theta(), {q});
633 } else {
634 circ.add_op<UnitID>(OpType::Rz, -node.theta(), {q});
635 }
636 break;
637 }
638 case 2: {
639 Complex x_coeff =
640 cast_coeff<quarter_turns_t, Complex>(tab.get_xrow(q).coeff);
641 if (x_coeff == 1.) {
642 circ.add_op<UnitID>(OpType::Rx, node.theta(), {q});
643 } else {
644 circ.add_op<UnitID>(OpType::Rx, -node.theta(), {q});
645 }
646 break;
647 }
648 default:
649 // support can't be Pauli::I
650 TKET_ASSERT(false);
651 }
652 bin.push_back(i);
653 }
654 if (bin.size() == 0) return false;
655 // sort the bin so we remove elements from back to front
656 std::sort(bin.begin(), bin.end(), std::greater<unsigned>());
657 for (const unsigned& index : bin) {
658 first_set.erase(first_set.begin() + index);
659 }
660 if (first_set.size() == 0) {
661 rotation_sets.erase(rotation_sets.begin());
662 return true;
663 }
664 return false;
665}

670static void pauli_exps_synthesis(
671 std::vector<std::vector<PauliExpNode>>& rotation_sets,
672 std::vector<TableauRowNode>& rows, UnitaryRevTableau& tab, Circuit& circ,
673 double discount_rate, double depth_weight, DepthTracker& depth_tracker) {
674 while (true) {
675 while (consume_available_rotations(
676 rotation_sets, tab, circ, depth_tracker)); // do nothing
677 if (rotation_sets.size() == 0) break;
678 std::vector<PauliExpNode>& first_set = rotation_sets[0];
679 // get nodes with min cost
680 std::vector<unsigned> min_nodes_indices = {0};
681 unsigned min_cost = first_set[0].tqe_cost();
682 for (unsigned i = 1; i < first_set.size(); i++) {
683 unsigned node_cost = first_set[i].tqe_cost();
684 if (node_cost == min_cost) {
685 min_nodes_indices.push_back(i);
686 } else if (node_cost < min_cost) {
687 min_nodes_indices = {i};
688 min_cost = node_cost;
689 }
690 }
691 std::set<TQE> tqe_candidates;
692 for (const unsigned& index : min_nodes_indices) {
693 std::vector<TQE> node_reducing_tqes = first_set[index].reduction_tqes();
694 tqe_candidates.insert(
695 node_reducing_tqes.begin(), node_reducing_tqes.end());
696 }
697 // for each tqe we compute costs which might subject to normalisation
698 std::map<TQE, std::vector<double>> tqe_candidates_cost;
699 for (const TQE& tqe : tqe_candidates) {
700 tqe_candidates_cost.insert(
701 {tqe,
702 {default_pauliexp_tqe_cost(discount_rate, rotation_sets, rows, tqe),
703 static_cast<double>(depth_tracker.gate_depth(
704 std::get<1>(tqe), std::get<2>(tqe)))}});
705 }
706 // select the best one
707 TQE selected_tqe = select_pauliexp_tqe(tqe_candidates_cost, depth_weight);
708 // apply TQE
709 apply_tqe_to_circ(selected_tqe, circ);
710 apply_tqe_to_tableau(selected_tqe, tab);
711 depth_tracker.add_2q_gate(
712 std::get<1>(selected_tqe), std::get<2>(selected_tqe));
713 for (std::vector<PauliExpNode>& rotation_set : rotation_sets) {
714 for (PauliExpNode& node : rotation_set) {
715 node.update(selected_tqe);
716 }
717 }
718 for (TableauRowNode& row : rows) {
719 row.update(selected_tqe);
720 }
721 }
722}

724// convert a Pauli exponential to a PauliExpNode
725static PauliExpNode get_node_from_exp(
726 const std::vector<Pauli>& paulis, const Expr& theta,
727 const qubit_vector_t& args, unsigned n, const UnitaryTableau& forward_tab,
728 const UnitaryRevTableau& tab) {
729 std::map<Qubit, Pauli> pauli_map;
730 for (unsigned i = 0; i < args.size(); i++) {
731 pauli_map.insert({args[i], paulis[i]});
732 }
733 // this has the effect of bringing the final clifford
734 // forward past the Pauli exponential
735 SpPauliStabiliser pstab =
736 forward_tab.get_row_product(SpPauliStabiliser(pauli_map));
737 Complex sign = cast_coeff<quarter_turns_t, Complex>(pstab.coeff);
738
739 std::vector<unsigned> support_vec;
740 for (unsigned i = 0; i < n; i++) {
741 SpPauliStabiliser zrow = tab.get_zrow(Qubit(i));
742 SpPauliStabiliser xrow = tab.get_xrow(Qubit(i));
743 bool z_supp = !zrow.commutes_with(pstab);
744 bool x_supp = !xrow.commutes_with(pstab);
745 if (!z_supp && !x_supp) {
746 support_vec.push_back(0);
747 } else if (!z_supp && x_supp) {
748 support_vec.push_back(1);
749 } else if (z_supp && !x_supp) {
750 support_vec.push_back(2);
751 } else if (z_supp && x_supp) {
752 support_vec.push_back(3);
753 }
754 }
755 return PauliExpNode(support_vec, sign.real() * theta);
756}

758// detect trivial pauli exps, if true then return the global phase
759static std::pair<bool, Expr> is_trivial_pauliexp(
760 const std::vector<Pauli>& paulis, const Expr& theta) {
761 if (static_cast<std::size_t>(std::count(
762 paulis.begin(), paulis.end(), Pauli::I)) == paulis.size()) {
763 // If all identity term
764 return {true, -theta / 2};
765 }
766 if (equiv_0(theta, 2)) {
767 if (equiv_0(theta, 4)) {
768 return {true, 0};
769 } else {
770 return {true, -1};
771 }
772 }
773 return {false, 0};
774}

776Circuit greedy_pauli_graph_synthesis(
777 const Circuit& circ, double discount_rate, double depth_weight) {
778 // c is the circuit we are trying to build
779 Circuit c(circ.all_qubits(), circ.all_bits());
780 std::optional<std::string> name = circ.get_name();
781 if (name != std::nullopt) {
782 c.set_name(name.value());
783 }
784 c.add_phase(circ.get_phase());
785 unit_map_t unit_map = c.flatten_registers();
786 Circuit measure_circ(c.n_qubits(), c.n_bits());
787 Circuit cliff(c.n_qubits());
788
789 // circuit used to iterate the original commands with flattened registers
790 Circuit circ_flat(circ);
791 circ_flat.flatten_registers();
792 std::vector<Command> commands = circ_flat.get_commands();
793 // extract the final clifford and the measurement circuits
794 for (const Command& cmd : commands) {
795 OpType optype = cmd.get_op_ptr()->get_type();
796 switch (optype) {
797 case OpType::Measure: {
798 measure_circ.add_op<UnitID>(OpType::Measure, cmd.get_args());
799 break;
800 }
801 default: {
802 if (optype == OpType::PauliExpBox ||
803 optype == OpType::PauliExpPairBox ||
804 optype == OpType::PauliExpCommutingSetBox)
805 break;
806 TKET_ASSERT(is_clifford_type(optype) && is_gate_type(optype));
807 cliff.add_op<UnitID>(optype, cmd.get_args());
808 }
809 }
810 }
811 std::vector<std::vector<PauliExpNode>> rotation_sets;
812 std::vector<TableauRowNode> rows;
813 // use forward Tableau to update the paulis by commuting the tableau to the
814 // front
815 UnitaryTableau forward_tab = circuit_to_unitary_tableau(cliff);
816 // Tableau used for tracking Cliffords throughout the synthesis
817 // TODO: this can be potentially made redundant
818 UnitaryRevTableau tab = circuit_to_unitary_rev_tableau(cliff).dagger();
819 unsigned n_qubits = c.n_qubits();
820 // extract the Pauli exps
821 for (const Command& cmd : commands) {
822 OpType optype = cmd.get_op_ptr()->get_type();
823 switch (optype) {
824 case OpType::PauliExpBox: {
825 const PauliExpBox& pbox =
826 static_cast<const PauliExpBox&>(*cmd.get_op_ptr());
827 const Expr phase = pbox.get_phase();
828 const std::vector<Pauli> paulis = pbox.get_paulis();
829 auto [trivial, global_phase] = is_trivial_pauliexp(paulis, phase);
830 if (trivial) {
831 c.add_phase(global_phase);
832 } else {
833 rotation_sets.push_back({get_node_from_exp(
834 paulis, phase, cmd.get_qubits(), n_qubits, forward_tab, tab)});
835 }
836 break;
837 }
838 case OpType::PauliExpPairBox: {
839 const PauliExpPairBox& pbox =
840 static_cast<const PauliExpPairBox&>(*cmd.get_op_ptr());
841 const auto [paulis1, paulis2] = pbox.get_paulis_pair();
842 const auto [phase1, phase2] = pbox.get_phase_pair();
843 auto [trivial1, global_phase1] = is_trivial_pauliexp(paulis1, phase1);
844 auto [trivial2, global_phase2] = is_trivial_pauliexp(paulis2, phase2);
845 std::vector<PauliExpNode> rotation_set;
846 if (trivial1) {
847 c.add_phase(global_phase1);
848 } else {
849 rotation_set.push_back(get_node_from_exp(
850 paulis1, phase1, cmd.get_qubits(), n_qubits, forward_tab, tab));
851 }
852 if (trivial2) {
853 c.add_phase(global_phase2);
854 } else {
855 rotation_set.push_back(get_node_from_exp(
856 paulis2, phase2, cmd.get_qubits(), n_qubits, forward_tab, tab));
857 }
858 if (!rotation_set.empty()) {
859 rotation_sets.push_back(rotation_set);
860 }
861 break;
862 }
863 case OpType::PauliExpCommutingSetBox: {
864 const PauliExpCommutingSetBox& pbox =
865 static_cast<const PauliExpCommutingSetBox&>(*cmd.get_op_ptr());
866 const std::vector<SymPauliTensor> gadgets = pbox.get_pauli_gadgets();
867 std::vector<PauliExpNode> rotation_set;
868 for (const SymPauliTensor& pt : gadgets) {
869 const std::vector<Pauli> paulis = pt.string;
870 const Expr phase = pt.coeff;
871 auto [trivial, global_phase] = is_trivial_pauliexp(paulis, phase);
872 if (trivial) {
873 c.add_phase(global_phase);
874 } else {
875 rotation_set.push_back(get_node_from_exp(
876 paulis, phase, cmd.get_qubits(), n_qubits, forward_tab, tab));
877 }
878 }
879 if (rotation_set.size() > 0) {
880 rotation_sets.push_back(rotation_set);
881 }
882 break;
883 }
884 default:
885 break;
886 }
887 }
888 // add identity TableauRowNodes
889 for (unsigned i = 0; i < n_qubits; i++) {
890 std::vector<unsigned> support_vec;
891 // identity rows
892 std::map<Qubit, Pauli> p;
893 std::map<Qubit, Pauli> q;
894 for (unsigned j = 0; j < n_qubits; j++) {
895 if (j == i) {
896 p.insert({Qubit(j), Pauli::Z});
897 q.insert({Qubit(j), Pauli::X});
898 } else {
899 p.insert({Qubit(j), Pauli::I});
900 q.insert({Qubit(j), Pauli::I});
901 }
902 }
903 SpPauliStabiliser stab_p(p);
904 SpPauliStabiliser stab_q(q);
905 for (unsigned row_index = 0; row_index < n_qubits; row_index++) {
906 SpPauliStabiliser zrow = tab.get_zrow(Qubit(row_index));
907 SpPauliStabiliser xrow = tab.get_xrow(Qubit(row_index));
908 bool lpx = !xrow.commutes_with(stab_p);
909 bool lpz = !zrow.commutes_with(stab_p);
910 bool lqx = !xrow.commutes_with(stab_q);
911 bool lqz = !zrow.commutes_with(stab_q);
912 support_vec.push_back(8 * lpx + 4 * lpz + 2 * lqx + lqz);
913 }
914 rows.push_back(TableauRowNode(support_vec));
915 }
916 DepthTracker depth_tracker(n_qubits);
917 // synthesise Pauli exps
918 pauli_exps_synthesis(
919 rotation_sets, rows, tab, c, discount_rate, depth_weight, depth_tracker);
920 // synthesise the tableau
921 tableau_row_nodes_synthesis(rows, tab, c, depth_weight, depth_tracker);
922 unit_map_t rev_unit_map;
923 for (const auto& pair : unit_map) {
924 rev_unit_map.insert({pair.second, pair.first});
925 }
926 c.append(measure_circ);
927 c.rename_units(rev_unit_map);
928 c.replace_SWAPs();
929 return c;
930}
